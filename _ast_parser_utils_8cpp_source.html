47 void RuleBody::conjunct(RuleBody&& other) {
48  std::vector<clause> res;
49 
50  for (const auto& clauseA : dnf) {
51  for (const auto& clauseB : other.dnf) {
52  // create a new clause in result
53  clause cur;
54 
55  // it is the concatenation of the two clauses
56  for (const auto& lit : clauseA) {
57  cur.emplace_back(lit.clone());
58  }
59  for (const auto& lit : clauseB) {
60  insert(cur, lit.clone());
61  }
62 
63  insert(res, std::move(cur));
64  }
65  }
66 
67  // update local dnf
68  dnf.swap(res);

79 std::vector<AstClause*> RuleBody::toClauseBodies() const {
80  // collect clause results
81  std::vector<AstClause*> bodies;
82  for (const clause& cur : dnf) {
83  bodies.push_back(new AstClause());
84  AstClause& clause = *bodies.back();
85 
86  for (const literal& lit : cur) {
87  // extract literal
88  AstLiteral* base = lit.atom->clone();
89  // negate if necessary
90  if (lit.negated) {
91  // negate
92  if (auto* atom = dynamic_cast<AstAtom*>(base)) {
93  base = new AstNegation(std::unique_ptr<AstAtom>(atom));
94  base->setSrcLoc(atom->getSrcLoc());
95  } else if (auto* cstr = dynamic_cast<AstConstraint*>(base)) {
96  negateConstraint(cstr);
97  }
98  }
99 
100  // add to result
101  clause.addToBody(std::unique_ptr<AstLiteral>(base));
102  }
103  }
104 
105  // done
106  return bodies;
107 }

199 void RuleBody::insert(std::vector<clause>& cnf, clause&& cls) {
200  for (const auto& cur : cnf) {
201  if (isSubsetOf(cur, cls)) {
202  return;
203  }
204  }
205  std::vector<clause> res;
206  for (auto& cur : cnf) {
207  if (!isSubsetOf(cls, cur)) {
208  res.push_back(std::move(cur));
209  }
210  }
211  res.swap(cnf);
212  cnf.push_back(std::move(cls));
213 }
