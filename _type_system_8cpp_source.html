116 /**
117  * A visitor for Types.
118  */
119 template <typename R>
120 struct TypeVisitor {
121  virtual ~TypeVisitor() = default;
122 
123  R operator()(const Type& type) const {
124  return visit(type);
125  }
126 
127  virtual R visit(const Type& type) const {
128  // check all kinds of types and dispatch
129  if (auto* t = dynamic_cast<const PredefinedType*>(&type)) {
130  return visitPredefinedType(*t);
131  }
132  if (auto* t = dynamic_cast<const SubsetType*>(&type)) {
133  return visitSubsetType(*t);
134  }
135  if (auto* t = dynamic_cast<const UnionType*>(&type)) {
136  return visitUnionType(*t);
137  }
138  if (auto* t = dynamic_cast<const RecordType*>(&type)) {
139  return visitRecordType(*t);
140  }
141  assert(false && "Unsupported type encountered!");
142  return R();
143  }
144 
145  virtual R visitPredefinedType(const PredefinedType& type) const {
146  return visitType(type);
147  }
148 
149  virtual R visitSubsetType(const SubsetType& type) const {
150  return visitType(type);
151  }
152 
153  virtual R visitUnionType(const UnionType& type) const {
154  return visitType(type);
155  }
156 
157  virtual R visitRecordType(const RecordType& type) const {
158  return visitType(type);
159  }
160 
161  virtual R visitType(const Type& /*type*/) const {
162  return R();
163  }
164 };

166 /**
167  * A visitor for types visiting each type only once (effectively breaking
168  * recursive cycles).
169  */
170 template <typename R>
171 class VisitOnceTypeVisitor : public TypeVisitor<R> {
172 protected:
173  mutable std::map<const Type*, R> seen;
174 
175 public:
176  R visit(const Type& type) const override {
177  auto pos = seen.find(&type);
178  if (pos != seen.end()) {
179  return pos->second;
180  }
181  auto& res = seen[&type]; // mark as seen
182  return res = TypeVisitor<R>::visit(type);
183  }
184 };

196 /**
197  * Determines whether the given type is a sub-type of the given root type.
198  */
199 bool isOfRootType(const Type& type, const Type& root) {
200  struct visitor : public VisitOnceTypeVisitor<bool> {
201  const Type& root;
202 
203  visitor(const Type& root) : root(root) {}
204 
205  bool visitPredefinedType(const PredefinedType& type) const override {
206  return type == root;
207  }
208  bool visitSubsetType(const SubsetType& type) const override {
209  return type == root || type.getBaseType() == root || isOfRootType(type.getBaseType(), root);
210  }
211  bool visitUnionType(const UnionType& type) const override {
212  if (type.getElementTypes().empty()) {
213  return false;
214  }
215  auto fit = [&](const Type* cur) { return visit(*cur); };
216  return all_of(type.getElementTypes(), fit);
217  }
218  bool visitType(const Type& /*unused*/) const override {
219  return false;
220  }
221  };
222 
223  return visitor(root).visit(type);
224 }

253 std::string getTypeQualifier(const Type& type) {
254  struct visitor : public VisitOnceTypeVisitor<std::string> {
255  std::string visitUnionType(const UnionType& type) const override {
256  std::string str = visitType(type);
257  str += "[";
258  bool first = true;
259  for (auto unionType : type.getElementTypes()) {
260  if (first) {
261  first = false;
262  } else {
263  str += ",";
264  }
265  str += visit(*unionType);
266  }
267  str += "]";
268  return str;
269  }

271  std::string visitRecordType(const RecordType& type) const override {
272  std::string str = visitType(type);
273  str += "{";
274  bool first = true;
275  for (auto field : type.getFields()) {
276  if (first) {
277  first = false;
278  } else {
279  str += ",";
280  }
281  str += field.name;
282  str += "#";
283  str += visit(field.type);
284  }
285  str += "}";
286  return str;
287  }

289  std::string visitType(const Type& type) const override {
290  std::string str;
291 
292  switch (getTypeAttribute(type)) {
293  case TypeAttribute::Signed:
294  str.append("i");
295  break;
296  case TypeAttribute::Unsigned:
297  str.append("u");
298  break;
299  case TypeAttribute::Float:
300  str.append("f");
301  break;
302  case TypeAttribute::Symbol:
303  str.append("s");
304  break;
305  case TypeAttribute::Record:
306  str.append("r");
307  break;
308  }
309  str.append(":");
310  str.append(toString(type.getName()));
311  seen[&type] = str;
312  return str;
313  }
314  };
315 
316  return visitor().visit(type);
317 }

371 bool isRecursiveType(const Type& type) {
372  struct visitor : public VisitOnceTypeVisitor<bool> {
373  const Type& trg;
374  visitor(const Type& trg) : trg(trg) {}
375  bool visit(const Type& type) const override {
376  if (trg == type) {
377  return true;
378  }
379  return VisitOnceTypeVisitor<bool>::visit(type);
380  }
381  bool visitUnionType(const UnionType& type) const override {
382  auto reachesTrg = [&](const Type* cur) { return this->visit(*cur); };
383  return any_of(type.getElementTypes(), reachesTrg);
384  }
385  bool visitRecordType(const RecordType& type) const override {
386  auto reachesTrg = [&](const RecordType::Field& cur) { return this->visit(cur.type); };
387  return any_of(type.getFields(), reachesTrg);
388  }
389  };
390 
391  // record types are recursive if they contain themselves
392  if (const auto* r = dynamic_cast<const RecordType*>(&type)) {
393  auto reachesOrigin = visitor(type);
394  return any_of(r->getFields(),
395  [&](const RecordType::Field& field) -> bool { return reachesOrigin(field.type); });
396  }
397 
398  return false;
399 }

401 bool isSubtypeOf(const Type& a, const Type& b) {
402  // make sure they are both in the same environment
403  auto& environment = a.getTypeEnvironment();
404  assert(environment.isType(a) && environment.isType(b));
405 
406  // first check - a type is a sub-type of itself
407  if (a == b) {
408  return true;
409  }
410 
411  // check for predefined types
412  if (b == environment.getNumberType()) {
413  return isNumberType(a);
414  }
415  if (b == environment.getSymbolType()) {
416  return isSymbolType(a);
417  }
418 
419  // check primitive type chains
420  if (isA<SubsetType>(a)) {
421  if (isSubtypeOf(as<SubsetType>(a).getBaseType(), b)) {
422  return true;
423  }
424  }
425 
426  // next - if b is a union type
427  if (isUnion(b)) {
428  return isSubType(a, as<UnionType>(b));
429  }
430 
431  // done
432  return false;
433 }

446 TypeSet getLeastCommonSupertypes(const Type& a, const Type& b) {
447  // make sure they are in the same type environment
448  assert(a.getTypeEnvironment().isType(a) && a.getTypeEnvironment().isType(b));
449 
450  // if they are equal it is easy
451  if (a == b) {
452  return TypeSet(a);
453  }
454 
455  // equally simple - check whether one is a sub-type of the other
456  if (isSubtypeOf(a, b)) {
457  return TypeSet(b);
458  }
459  if (isSubtypeOf(b, a)) {
460  return TypeSet(a);
461  }
462 
463  // harder: no obvious relation => hard way
464  TypeSet superTypes;
465  TypeSet all = a.getTypeEnvironment().getAllTypes();
466  for (const Type& cur : all) {
467  if (isSubtypeOf(a, cur) && isSubtypeOf(b, cur)) {
468  superTypes.insert(cur);
469  }
470  }
471 
472  // filter out non-least super types
473  TypeSet res;
474  for (const Type& cur : superTypes) {
475  bool least = !any_of(superTypes, [&](const Type& t) { return t != cur && isSubtypeOf(t, cur); });
476  if (least) {
477  res.insert(cur);
478  }
479  }
480 
481  return res;
482 }

484 TypeSet getLeastCommonSupertypes(const TypeSet& set) {
485  // handle the empty set
486  if (set.empty()) {
487  return set;
488  }
489 
490  // handle the all set => empty set (since no common super-type)
491  if (set.isAll()) {
492  return TypeSet();
493  }
494 
495  TypeSet res;
496  auto it = set.begin();
497  res.insert(*it);
498  ++it;
499 
500  // refine sub-set step by step
501  for (; it != set.end(); ++it) {
502  TypeSet tmp;
503  for (const Type& cur : res) {
504  tmp.insert(getLeastCommonSupertypes(cur, *it));
505  }
506  res = tmp;
507  }
508 
509  // done
510  return res;
511 }

540 TypeSet getGreatestCommonSubtypes(const Type& a, const Type& b) {
541  // make sure they are in the same type environment
542  assert(a.getTypeEnvironment().isType(a) && a.getTypeEnvironment().isType(b));
543 
544  // if they are equal it is easy
545  if (a == b) {
546  return TypeSet(a);
547  }
548 
549  // equally simple - check whether one is a sub-type of the other
550  if (isSubtypeOf(a, b)) {
551  return TypeSet(a);
552  }
553  if (isSubtypeOf(b, a)) {
554  return TypeSet(b);
555  }
556 
557  // last option: if both are unions with common sub-types
558  TypeSet res;
559  if (isUnion(a) && isUnion(b)) {
560  // collect common sub-types of union types
561 
562  struct collector : public TypeVisitor<void> {
563  const Type& b;
564  TypeSet& res;
565  collector(const Type& b, TypeSet& res) : b(b), res(res) {}
566  void visit(const Type& type) const override {
567  if (isSubtypeOf(type, b)) {
568  res.insert(type);
569  } else {
570  TypeVisitor<void>::visit(type);
571  }
572  }
573  void visitUnionType(const UnionType& type) const override {
574  for (const auto& cur : type.getElementTypes()) {
575  visit(*cur);
576  }
577  }
578  };
579 
580  // collect all common sub-types
581  collector(b, res).visit(a);
582  }
583 
584  // otherwise there is no common super type
585  return res;
586 }
