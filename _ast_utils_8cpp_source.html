83 std::set<const AstRelation*> getBodyRelations(const AstClause* clause, const AstProgram* program) {
84  std::set<const AstRelation*> bodyRelations;
85  for (const auto& lit : clause->getBodyLiterals()) {
86  visitDepthFirst(
87  *lit, [&](const AstAtom& atom) { bodyRelations.insert(getAtomRelation(&atom, program)); });
88  }
89  for (const auto& arg : clause->getHead()->getArguments()) {
90  visitDepthFirst(
91  *arg, [&](const AstAtom& atom) { bodyRelations.insert(getAtomRelation(&atom, program)); });
92  }
93  return bodyRelations;
94 }

96 size_t getClauseNum(const AstProgram* program, const AstClause* clause) {
97  // TODO (azreika): This number might change between the provenance transformer and the AST->RAM
98  // translation. Might need a better way to assign IDs to clauses... (see PR #1288).
99  const AstRelation* rel = getRelation(*program, clause->getHead()->getQualifiedName());
100  assert(rel != nullptr && "clause relation does not exist");
101 
102  size_t clauseNum = 1;
103  for (const auto* cur : getClauses(*program, *rel)) {
104  bool isFact = cur->getBodyLiterals().empty();
105  if (cur == clause) {
106  return isFact ? 0 : clauseNum;
107  }
108 
109  if (!isFact) {
110  clauseNum++;
111  }
112  }
113  assert(false && "clause does not exist");
114 }

159 bool isFact(const AstClause& clause) {
160  // there must be a head
161  if (clause.getHead() == nullptr) {
162  return false;
163  }
164  // there must not be any body clauses
165  if (!clause.getBodyLiterals().empty()) {
166  return false;
167  }
168 
169  // and there are no aggregates
170  bool hasAggregates = false;
171  visitDepthFirst(*clause.getHead(), [&](const AstAggregator&) { hasAggregates = true; });
172  return !hasAggregates;
173 }

189 AstClause* reorderAtoms(const AstClause* clause, const std::vector<unsigned int>& newOrder) {
190  // Find all atom positions
191  std::vector<unsigned int> atomPositions;
192  std::vector<AstLiteral*> bodyLiterals = clause->getBodyLiterals();
193  for (unsigned int i = 0; i < bodyLiterals.size(); i++) {
194  if (dynamic_cast<AstAtom*>(bodyLiterals[i]) != nullptr) {
195  atomPositions.push_back(i);
196  }
197  }
198 
199  // Validate given order
200  assert(newOrder.size() == atomPositions.size());
201  std::vector<unsigned int> nopOrder;
202  for (unsigned int i = 0; i < atomPositions.size(); i++) {
203  nopOrder.push_back(i);
204  }
205  assert(std::is_permutation(nopOrder.begin(), nopOrder.end(), newOrder.begin()));
206 
207  // Create a new clause with the given atom order, leaving the rest unchanged
208  AstClause* newClause = cloneHead(clause);
209  unsigned int currentAtom = 0;
210  for (unsigned int currentLiteral = 0; currentLiteral < bodyLiterals.size(); currentLiteral++) {
211  AstLiteral* literalToAdd = bodyLiterals[currentLiteral];
212  if (dynamic_cast<AstAtom*>(literalToAdd) != nullptr) {
213  // Atoms should be reordered
214  literalToAdd = bodyLiterals[atomPositions[newOrder[currentAtom++]]];
215  }
216  newClause->addToBody(std::unique_ptr<AstLiteral>(literalToAdd->clone()));
217  }
218 
219  return newClause;
220 }

222 void negateConstraint(AstConstraint* constraint) {
223  assert(nullptr != dynamic_cast<AstConstraint*>(constraint) && "not a constraint object");
224  if (auto* bcstr = dynamic_cast<AstBooleanConstraint*>(constraint)) {
225  bcstr->set(!bcstr->isTrue());
226  } else if (auto* cstr = dynamic_cast<AstBinaryConstraint*>(constraint)) {
227  cstr->setOperator(souffle::negatedConstraintOp(cstr->getOperator()));
228  } else {
229  assert(false && "Unknown ast-constraint type");
230  }
231 }
