53 /**
54  * An implementation of a meet operation between sets of types computing
55  * the set of pair-wise greatest common subtypes.
56  */
57 struct sub_type {
58  bool operator()(TypeSet& a, const TypeSet& b) const {
59  // compute result set
60  TypeSet res = getGreatestCommonSubtypes(a, b);
61 
62  // check whether a should change
63  if (res == a) {
64  return false;
65  }
66 
67  // update a
68  a = res;
69  return true;
70  }
71 };

82 /**
83  * The type lattice forming the property space for the Type analysis. The
84  * value set is given by sets of types and the meet operator is based on the
85  * pair-wise computation of greatest common subtypes. Correspondingly, the
86  * bottom element has to be the set of all types.
87  */
88 struct type_lattice : public property_space<TypeSet, sub_type, all_type_factory> {};

104 /**
105  * A constraint factory ensuring that all the types associated to the variable
106  * a are subtypes of type b.
107  */
108 TypeConstraint isSubtypeOf(const TypeVar& a, const Type& b) {
109  struct C : public Constraint<TypeVar> {
110  TypeVar a;
111  const Type& b;
112 
113  C(TypeVar a, const Type& b) : a(std::move(a)), b(b) {}
114 
115  bool update(Assignment<TypeVar>& ass) const override {
116  // get current value of variable a
117  TypeSet& s = ass[a];
118 
119  // remove all types that are not sub-types of b
120  if (s.isAll()) {
121  s = TypeSet(b);
122  return true;
123  }
124 
125  TypeSet res;
126  for (const Type& t : s) {
127  res.insert(getGreatestCommonSubtypes(t, b));
128  }
129 
130  // check whether there was a change
131  if (res == s) {
132  return false;
133  }
134  s = res;
135  return true;
136  }
137 
138  void print(std::ostream& out) const override {
139  out << a << " <: " << b.getName();
140  }
141  };
142 
143  return std::make_shared<C>(a, b);
144 }

146 /**
147  * A constraint factory ensuring that all the types associated to the variable
148  * are subtypes of some type in the provided set (values)
149  *
150  * Values can't be all.
151  */
152 TypeConstraint hasSuperTypeInSet(const TypeVar& var, TypeSet values) {
153  struct C : public Constraint<TypeVar> {
154  TypeVar var;
155  TypeSet values;
156 
157  C(TypeVar var, TypeSet values) : var(std::move(var)), values(std::move(values)) {}
158 
159  bool update(Assignment<TypeVar>& assigment) const override {
160  // get current value of variable a
161  TypeSet& assigments = assigment[var];
162 
163  // remove all types that are not sub-types of b
164  if (assigments.isAll()) {
165  assigments = values;
166  return true;
167  }
168 
169  TypeSet newAssigments;
170  for (const Type& type : assigments) {
171  bool existsSuperTypeInValues = any_of(values.begin(), values.end(),
172  [&type](const Type& value) { return isSubtypeOf(type, value); });
173  if (existsSuperTypeInValues) {
174  newAssigments.insert(type);
175  }
176  }
177 
178  // check whether there was a change
179  if (newAssigments == assigments) {
180  return false;
181  }
182  assigments = newAssigments;
183  return true;
184  }
185 
186  void print(std::ostream& out) const override {
187  out << "∃ t ∈ " << values << ": " << var << " <: t";
188  }
189  };
190 
191  return std::make_shared<C>(var, values);
192 }

194 /**
195  * Ensure that types of left and right have the same base types.
196  */
197 TypeConstraint subtypesOfTheSameBaseType(const TypeVar& left, const TypeVar& right) {
198  struct C : public Constraint<TypeVar> {
199  TypeVar left;
200  TypeVar right;
201 
202  C(TypeVar left, TypeVar right) : left(std::move(left)), right(std::move(right)) {}
203 
204  bool update(Assignment<TypeVar>& assigment) const override {
205  // get current value of variable a
206  TypeSet& assigmentsLeft = assigment[left];
207  TypeSet& assigmentsRight = assigment[right];
208 
209  // Base types common to left and right variables.
210  TypeSet baseTypes;
211 
212  // Base types present in left/right variable.
213  TypeSet baseTypesLeft;
214  TypeSet baseTypesRight;
215 
216  // Iterate over possible types extracting base types.
217  // Left
218  if (!assigmentsLeft.isAll()) {
219  for (const auto& type : assigmentsLeft) {
220  // Predefined type is always a base type.
221  if (dynamic_cast<const PredefinedType*>(&type) != nullptr) {
222  baseTypesLeft.insert(type);
223  } else if (auto* primitive = dynamic_cast<const SubsetType*>(&type)) {
224  baseTypesLeft.insert(primitive->getBaseType());
225  }
226  }
227  }
228  // Right
229  if (!assigmentsRight.isAll()) {
230  for (const auto& type : assigmentsRight) {
231  // Predefined type is always a base type.
232  if (dynamic_cast<const PredefinedType*>(&type) != nullptr) {
233  baseTypesRight.insert(type);
234  } else if (auto* primitive = dynamic_cast<const SubsetType*>(&type)) {
235  baseTypesRight.insert(primitive->getBaseType());
236  }
237  }
238  }
239 
240  baseTypes = TypeSet::intersection(baseTypesLeft, baseTypesRight);
241 
242  TypeSet resultLeft;
243  TypeSet resultRight;
244 
245  // Handle all
246  if (assigmentsLeft.isAll() && assigmentsRight.isAll()) {
247  return false;
248  }
249 
250  // If left xor right is all, assign base types of the other side as possible values.
251  if (assigmentsLeft.isAll()) {
252  assigmentsLeft = baseTypesRight;
253  return true;
254  }
255  if (assigmentsRight.isAll()) {
256  assigmentsRight = baseTypesLeft;
257  return true;
258  }
259 
260  // Allow types if they are subtypes of any of the common base types.
261  for (const Type& type : assigmentsLeft) {
262  bool isSubtypeOfCommonBaseType = any_of(baseTypes.begin(), baseTypes.end(),
263  [&type](const Type& baseType) { return isSubtypeOf(type, baseType); });
264  if (isSubtypeOfCommonBaseType) {
265  resultLeft.insert(type);
266  }
267  }
268 
269  for (const Type& type : assigmentsRight) {
270  bool isSubtypeOfCommonBaseType = any_of(baseTypes.begin(), baseTypes.end(),
271  [&type](const Type& baseType) { return isSubtypeOf(type, baseType); });
272  if (isSubtypeOfCommonBaseType) {
273  resultRight.insert(type);
274  }
275  }
276 
277  // check whether there was a change
278  if (resultLeft == assigmentsLeft && resultRight == assigmentsRight) {
279  return false;
280  }
281  assigmentsLeft = resultLeft;
282  assigmentsRight = resultRight;
283  return true;
284  }
285  //
286  void print(std::ostream& out) const override {
287  out << "∃ t : (" << left << " <: t)"
288  << " ∧ "
289  << "(" << right << " <: t)"
290  << " where t is a base type";
291  }
292  };
293 
294  return std::make_shared<C>(left, right);
295 }

297 /**
298  * A constraint factory ensuring that all the types associated to the variable
299  * a are subtypes of type b.
300  */
301 TypeConstraint isSupertypeOf(const TypeVar& a, const Type& b) {
302  struct C : public Constraint<TypeVar> {
303  TypeVar a;
304  const Type& b;
305  mutable bool repeat;
306 
307  C(TypeVar a, const Type& b) : a(std::move(a)), b(b), repeat(true) {}
308 
309  bool update(Assignment<TypeVar>& ass) const override {
310  // don't continually update super type constraints
311  if (!repeat) {
312  return false;
313  }
314  repeat = false;
315 
316  // get current value of variable a
317  TypeSet& s = ass[a];
318 
319  // remove all types that are not super-types of b
320  if (s.isAll()) {
321  s = TypeSet(b);
322  return true;
323  }
324 
325  TypeSet res;
326  for (const Type& t : s) {
327  res.insert(getLeastCommonSupertypes(t, b));
328  }
329 
330  // check whether there was a change
331  if (res == s) {
332  return false;
333  }
334  s = res;
335  return true;
336  }
337 
338  void print(std::ostream& out) const override {
339  out << a << " >: " << b.getName();
340  }
341  };
342 
343  return std::make_shared<C>(a, b);
344 }

346 TypeConstraint isSubtypeOfComponent(const TypeVar& a, const TypeVar& b, int index) {
347  struct C : public Constraint<TypeVar> {
348  TypeVar a;
349  TypeVar b;
350  unsigned index;
351 
352  C(TypeVar a, TypeVar b, int index) : a(std::move(a)), b(std::move(b)), index(index) {}
353 
354  bool update(Assignment<TypeVar>& ass) const override {
355  // get list of types for b
356  const TypeSet& recs = ass[b];
357 
358  // if it is (not yet) constrainted => skip
359  if (recs.isAll()) {
360  return false;
361  }
362 
363  // compute new types for a and b
364  TypeSet typesA;
365  TypeSet typesB;
366 
367  // iterate through types of b
368  for (const Type& t : recs) {
369  // only retain records
370  if (!isRecordType(t)) {
371  continue;
372  }
373  const auto& rec = static_cast<const RecordType&>(t);
374 
375  // of proper size
376  if (rec.getFields().size() <= index) {
377  continue;
378  }
379 
380  // this is a valid type for b
381  typesB.insert(rec);
382 
383  // and its corresponding field for a
384  typesA.insert(rec.getFields()[index].type);
385  }
386 
387  // combine with current types assigned to a
388  typesA = getGreatestCommonSubtypes(ass[a], typesA);
389 
390  // update values
391  bool changed = false;
392  if (recs != typesB) {
393  ass[b] = typesB;
394  changed = true;
395  }
396 
397  if (ass[a] != typesA) {
398  ass[a] = typesA;
399  changed = true;
400  }
401 
402  // done
403  return changed;
404  }
405 
406  void print(std::ostream& out) const override {
407  out << a << " <: " << b << "::" << index;
408  }
409  };
410 
411  return std::make_shared<C>(a, b, index);
412 }

415 /* Return a new clause with type-annotated variables */
416 AstClause* createAnnotatedClause(
417  const AstClause* clause, const std::map<const AstArgument*, TypeSet> argumentTypes) {
418  // Annotates each variable with its type based on a given type analysis result
419  struct TypeAnnotator : public AstNodeMapper {
420  const std::map<const AstArgument*, TypeSet>& types;
421 
422  TypeAnnotator(const std::map<const AstArgument*, TypeSet>& types) : types(types) {}
423 
424  std::unique_ptr<AstNode> operator()(std::unique_ptr<AstNode> node) const override {
425  if (auto* var = dynamic_cast<AstVariable*>(node.get())) {
426  std::stringstream newVarName;
427  newVarName << var->getName() << "&isin;" << types.find(var)->second;
428  return std::make_unique<AstVariable>(newVarName.str());
429  } else if (auto* var = dynamic_cast<AstUnnamedVariable*>(node.get())) {
430  std::stringstream newVarName;
431  newVarName << "_"
432  << "&isin;" << types.find(var)->second;
433  return std::make_unique<AstVariable>(newVarName.str());
434  }
435  node->apply(*this);
436  return node;
437  }
438  };
439 
440  /* Note:
441  * Because the type of each argument is stored in the form [address -> type-set],
442  * the type-analysis result does not immediately apply to the clone due to differing
443  * addresses.
444  * Two ways around this:
445  * (1) Perform the type-analysis again for the cloned clause
446  * (2) Keep track of the addresses of equivalent arguments in the cloned clause
447  * Method (2) was chosen to avoid having to recompute the analysis each time.
448  */
449  AstClause* annotatedClause = clause->clone();
450 
451  // Maps x -> y, where x is the address of an argument in the original clause, and y
452  // is the address of the equivalent argument in the clone.
453  std::map<const AstArgument*, const AstArgument*> memoryMap;
454 
455  std::vector<const AstArgument*> originalAddresses;
456  visitDepthFirst(*clause, [&](const AstArgument& arg) { originalAddresses.push_back(&arg); });
457 
458  std::vector<const AstArgument*> cloneAddresses;
459  visitDepthFirst(*annotatedClause, [&](const AstArgument& arg) { cloneAddresses.push_back(&arg); });
460 
461  assert(cloneAddresses.size() == originalAddresses.size());
462 
463  for (size_t i = 0; i < originalAddresses.size(); i++) {
464  memoryMap[originalAddresses[i]] = cloneAddresses[i];
465  }
466 
467  // Map the types to the clause clone
468  std::map<const AstArgument*, TypeSet> cloneArgumentTypes;
469  for (auto& pair : argumentTypes) {
470  cloneArgumentTypes[memoryMap[pair.first]] = pair.second;
471  }
472 
473  // Create the type-annotated clause
474  TypeAnnotator annotator(cloneArgumentTypes);
475  annotatedClause->apply(annotator);
476  return annotatedClause;
477 }

479 void TypeAnalysis::run(const AstTranslationUnit& translationUnit) {
480  // Check if debugging information is being generated and note where logs should be sent
481  std::ostream* debugStream = nullptr;
482  if (Global::config().has("debug-report") || Global::config().get("show") == "type-analysis") {
483  debugStream = &analysisLogs;
484  }
485  const auto& program = *translationUnit.getProgram();
486  auto* typeEnvAnalysis = translationUnit.getAnalysis<TypeEnvironmentAnalysis>();
487  for (const AstRelation* rel : translationUnit.getProgram()->getRelations()) {
488  for (const AstClause* clause : getClauses(program, *rel)) {
489  // Perform the type analysis
490  std::map<const AstArgument*, TypeSet> clauseArgumentTypes =
491  analyseTypes(typeEnvAnalysis->getTypeEnvironment(), *clause, translationUnit.getProgram(),
492  debugStream);
493  argumentTypes.insert(clauseArgumentTypes.begin(), clauseArgumentTypes.end());
494 
495  if (debugStream != nullptr) {
496  // Store an annotated clause for printing purposes
497  AstClause* annotatedClause = createAnnotatedClause(clause, clauseArgumentTypes);
498  annotatedClauses.emplace_back(annotatedClause);
499  }
500  }
501  }
502 }

517 std::map<const AstArgument*, TypeSet> TypeAnalysis::analyseTypes(
518  const TypeEnvironment& env, const AstClause& clause, const AstProgram* program, std::ostream* logs) {
519  struct Analysis : public AstConstraintAnalysis<TypeVar> {
520  const TypeEnvironment& env;
521  const AstProgram* program;
522  std::set<const AstAtom*> negated;
523 
524  Analysis(const TypeEnvironment& env, const AstProgram* program) : env(env), program(program) {}
525 
526  // predicate
527  void visitAtom(const AstAtom& atom) override {
528  // get relation
529  auto rel = getAtomRelation(&atom, program);
530  if (rel == nullptr) {
531  return; // error in input program
532  }
533 
534  auto atts = rel->getAttributes();
535  auto args = atom.getArguments();
536  if (atts.size() != args.size()) {
537  return; // error in input program
538  }
539 
540  // set upper boundary of argument types
541  for (unsigned i = 0; i < atts.size(); i++) {
542  const auto& typeName = atts[i]->getTypeName();
543  if (env.isType(typeName)) {
544  // check whether atom is not negated
545  if (negated.find(&atom) == negated.end()) {
546  addConstraint(isSubtypeOf(getVar(args[i]), env.getType(typeName)));
547  } else {
548  addConstraint(isSupertypeOf(getVar(args[i]), env.getType(typeName)));
549  }
550  }
551  }
552  }
553 
554  // negations need to be skipped
555  void visitNegation(const AstNegation& cur) override {
556  // add nested atom to black-list
557  negated.insert(cur.getAtom());
558  }
559 
560  // symbol
561  void visitStringConstant(const AstStringConstant& cnst) override {
562  addConstraint(isSubtypeOf(getVar(cnst), env.getSymbolType()));
563  }
564 
565  // Numeric constant
566  void visitNumericConstant(const AstNumericConstant& constant) override {
567  TypeSet possibleTypes;
568 
569  // Check if the type is given.
570  if (constant.getType().has_value()) {
571  switch (*constant.getType()) {
572  // Insert a type, but only after checking that parsing is possible.
573  case AstNumericConstant::Type::Int:
574  if (canBeParsedAsRamSigned(constant.getConstant())) {
575  possibleTypes.insert(env.getNumberType());
576  }
577  break;
578  case AstNumericConstant::Type::Uint:
579  if (canBeParsedAsRamUnsigned(constant.getConstant())) {
580  possibleTypes.insert(env.getUnsignedType());
581  }
582  break;
583  case AstNumericConstant::Type::Float:
584  if (canBeParsedAsRamFloat(constant.getConstant())) {
585  possibleTypes.insert(env.getFloatType());
586  }
587  break;
588  }
589  // Else: all numeric types that can be parsed are valid.
590  } else {
591  if (canBeParsedAsRamSigned(constant.getConstant())) {
592  possibleTypes.insert(env.getNumberType());
593  }
594 
595  if (canBeParsedAsRamUnsigned(constant.getConstant())) {
596  possibleTypes.insert(env.getUnsignedType());
597  }
598 
599  if (canBeParsedAsRamFloat(constant.getConstant())) {
600  possibleTypes.insert(env.getFloatType());
601  }
602  }
603 
604  addConstraint(hasSuperTypeInSet(getVar(constant), possibleTypes));
605  }
606 
607  // binary constraint
608  void visitBinaryConstraint(const AstBinaryConstraint& rel) override {
609  auto lhs = getVar(rel.getLHS());
610  auto rhs = getVar(rel.getRHS());
611  addConstraint(isSubtypeOf(lhs, rhs));
612  addConstraint(isSubtypeOf(rhs, lhs));
613  }
614 
615  // intrinsic functor
616  void visitFunctor(const AstFunctor& fun) override {
617  auto functorVar = getVar(fun);
618 
619  // In polymorphic case
620  // We only require arguments to share a base type with a return type.
621  // (instead of, for example, requiring them to be of the same type)
622  // This approach is related to old type semantics
623  // See #1296 and tests/semantic/type_system4
624  if (auto intrinsicFunctor = dynamic_cast<const AstIntrinsicFunctor*>(&fun)) {
625  if (isOverloadedFunctor(intrinsicFunctor->getFunction())) {
626  for (auto* argument : intrinsicFunctor->getArguments()) {
627  auto argumentVar = getVar(argument);
628  addConstraint(subtypesOfTheSameBaseType(argumentVar, functorVar));
629  }
630 
631  return;
632  }
633  }
634 
635  try {
636  fun.getReturnType();
637  } catch (std::bad_optional_access& e) {
638  return;
639  }
640 
641  // add a constraint for the return type of the functor
642  switch (fun.getReturnType()) {
643  case TypeAttribute::Signed:
644  addConstraint(isSubtypeOf(functorVar, env.getNumberType()));
645  break;
646  case TypeAttribute::Float:
647  addConstraint(isSubtypeOf(functorVar, env.getFloatType()));
648  break;
649  case TypeAttribute::Unsigned:
650  addConstraint(isSubtypeOf(functorVar, env.getUnsignedType()));
651  break;
652  case TypeAttribute::Symbol:
653  addConstraint(isSubtypeOf(functorVar, env.getSymbolType()));
654  break;
655  default:
656  assert(false && "Invalid return type");
657  }
658 
659  // Special case
660  if (auto intrFun = dynamic_cast<const AstIntrinsicFunctor*>(&fun)) {
661  if (intrFun->getFunction() == FunctorOp::ORD) {
662  return;
663  }
664  }
665  size_t i = 0;
666  for (auto arg : fun.getArguments()) {
667  auto argumentVar = getVar(arg);
668  switch (fun.getArgType(i)) {
669  case TypeAttribute::Signed:
670  addConstraint(isSubtypeOf(argumentVar, env.getNumberType()));
671  break;
672  case TypeAttribute::Float:
673  addConstraint(isSubtypeOf(argumentVar, env.getFloatType()));
674  break;
675  case TypeAttribute::Unsigned:
676  addConstraint(isSubtypeOf(argumentVar, env.getUnsignedType()));
677  break;
678  case TypeAttribute::Symbol:
679  addConstraint(isSubtypeOf(argumentVar, env.getSymbolType()));
680  break;
681  default:
682  assert(false && "Invalid argument type");
683  }
684  ++i;
685  }
686  }
687  // counter
688  void visitCounter(const AstCounter& counter) override {
689  // this value must be a number value
690  addConstraint(isSubtypeOf(getVar(counter), env.getNumberType()));
691  }
692 
693  // components of records
694  void visitRecordInit(const AstRecordInit& init) override {
695  // link element types with sub-values
696  auto rec = getVar(init);
697  int i = 0;
698 
699  for (const AstArgument* value : init.getArguments()) {
700  addConstraint(isSubtypeOfComponent(getVar(value), rec, i++));
701  }
702  }
703 
704  // visit aggregates
705  void visitAggregator(const AstAggregator& agg) override {
706  if (agg.getOperator() == AggregateOp::COUNT) {
707  addConstraint(isSubtypeOf(getVar(agg), env.getNumberType()));
708  } else if (agg.getOperator() == AggregateOp::MEAN) {
709  addConstraint(isSubtypeOf(getVar(agg), env.getFloatType()));
710  } else {
711  addConstraint(hasSuperTypeInSet(getVar(agg), env.getNumericTypes()));
712  }
713 
714  // If there is a target expression - it should be of the same type as the aggregator.
715  if (auto expr = agg.getTargetExpression()) {
716  addConstraint(isSubtypeOf(getVar(expr), getVar(agg)));
717  addConstraint(isSubtypeOf(getVar(agg), getVar(expr)));
718  }
719  }
720  };
721
