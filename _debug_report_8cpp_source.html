24 void DebugReportSection::printIndex(std::ostream& out) const {
25  out << "<a href=\"#" << id << "\">" << title << "</a>\n";
26  out << "<ul>\n";
27  bool isLeaf = true;
28  for (const DebugReportSection& subsection : subsections) {
29  if (subsection.hasSubsections()) {
30  isLeaf = false;
31  break;
32  }
33  }
34  for (const DebugReportSection& subsection : subsections) {
35  if (isLeaf) {
36  out << "<li class='leaf'>";
37  } else {
38  out << "<li>";
39  }
40  subsection.printIndex(out);
41  out << "</li>";
42  }
43  out << "</ul>\n";
44 }
