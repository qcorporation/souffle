32 /**
33  * A component type is
34  *
35  * name < Type1, Type2, ... >
36  *
37  * where name is the component name and < Type, Type, ... > is a
38  * list of component type parameters (either actual or formal).
39  */
40 class AstComponentType : public AstNode {
41 public:
42  AstComponentType(
43  std::string name = "", std::vector<AstQualifiedName> params = std::vector<AstQualifiedName>())
44  : name(std::move(name)), typeParams(std::move(params)) {}
45 
46  /** get component name */
47  const std::string& getName() const {
48  return name;
49  }
50 
51  /** set component name */
52  void setName(const std::string& n) {
53  name = n;
54  }
55 
56  /** get component type parameters */
57  const std::vector<AstQualifiedName>& getTypeParameters() const {
58  return typeParams;
59  }
60 
61  /** set component type parameters */
62  void setTypeParameters(const std::vector<AstQualifiedName>& params) {
63  typeParams = params;
64  }
65 
66  AstComponentType* clone() const override {
67  auto* res = new AstComponentType(name, typeParams);
68  res->setSrcLoc(getSrcLoc());
69  return res;
70  }
71 
72 protected:
73  void print(std::ostream& os) const override {
74  os << name;
75  if (!typeParams.empty()) {
76  os << "<" << join(typeParams, ",") << ">";
77  }
78  }
79 
80  bool equal(const AstNode& node) const override {
81  const auto& other = static_cast<const AstComponentType&>(node);
82  return name == other.name && typeParams == other.typeParams;
83  }
84 
85 private:
86  /** component name */
87  std::string name;
88 
89  /** component type parameters */
90  std::vector<AstQualifiedName> typeParams;
91 };
