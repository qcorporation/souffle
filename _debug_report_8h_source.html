82 /**
83  * Class representing a HTML report, consisting of a list of sections.
84  */
85 class DebugReport {
86 public:
87  ~DebugReport() {
88  while (!currentSubsections.empty()) {
89  endSection("forced-closed", "Forcing end of unknown section");
90  }
91  if (!empty()) {
92  std::ofstream debugReportStream(Global::config().get("debug-report"));
93  debugReportStream << *this;
94  }
95  }
96  bool empty() const {
97  return sections.empty();
98  }
99 
100  void addSection(const DebugReportSection& section) {
101  if (!currentSubsections.empty()) {
102  currentSubsections.top().emplace_back(section);
103  } else {
104  sections.emplace_back(section);
105  }
106  }
107 
108  void addSection(std::string id, std::string title, std::string code) {
109  std::stringstream codeHTML;
110  std::string escapedCode = std::move(code);
111  while (true) {
112  size_t i = escapedCode.find("<");
113  if (i == std::string::npos) {
114  break;
115  }
116  escapedCode.replace(i, 1, "&lt;");
117  }
118  codeHTML << "<pre>" << escapedCode << "</pre>\n";
119  if (!currentSubsections.empty()) {
120  currentSubsections.top().emplace_back(std::move(id), std::move(title), codeHTML.str());
121  } else {
122  sections.emplace_back(std::move(id), std::move(title), codeHTML.str());
123  }
124  }
125 
126  void startSection() {
127  currentSubsections.emplace();
128  }
129 
130  void endSection(std::string currentSectionName, std::string currentSectionTitle) {
131  auto subsections = currentSubsections.top();
132  currentSubsections.pop();
133  addSection(DebugReportSection(currentSectionName, currentSectionTitle, subsections, ""));
134  }

139  * followed by the content of each section.
140  */
141  void print(std::ostream& out) const;
