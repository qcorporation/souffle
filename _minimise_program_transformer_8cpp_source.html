38 /**
39  * Extract valid permutations from a given permutation matrix of valid moves.
40  */
41 std::vector<std::vector<unsigned int>> extractPermutations(
42  const std::vector<std::vector<unsigned int>>& permutationMatrix) {
43  size_t clauseSize = permutationMatrix.size();
44  // keep track of the possible end-positions of each atom in the first clause
45  std::vector<std::vector<unsigned int>> validMoves;
46  for (size_t i = 0; i < clauseSize; i++) {
47  std::vector<unsigned int> currentRow;
48  for (size_t j = 0; j < clauseSize; j++) {
49  if (permutationMatrix[i][j] == 1) {
50  currentRow.push_back(j);
51  }
52  }
53  validMoves.push_back(currentRow);
54  }
55 
56  // extract the possible permutations, DFS style
57  std::vector<std::vector<unsigned int>> permutations;
58  std::vector<unsigned int> seen(clauseSize);
59  std::vector<unsigned int> currentPermutation;
60  std::stack<std::vector<unsigned int>> todoStack;
61 
62  todoStack.push(validMoves[0]);
63 
64  size_t currentIdx = 0;
65  while (!todoStack.empty()) {
66  if (currentIdx == clauseSize) {
67  // permutation is complete
68  permutations.push_back(currentPermutation);
69 
70  if (currentIdx == 0) {
71  // already at starting position, so no more permutations possible
72  break;
73  }
74 
75  // undo the last number added to the permutation
76  currentIdx--;
77  seen[currentPermutation[currentIdx]] = 0;
78  currentPermutation.pop_back();
79 
80  // see if we can pick up other permutations
81  continue;
82  }
83 
84  // pull out the possibilities for the current point of the permutation
85  std::vector<unsigned int> possibilities = todoStack.top();
86  todoStack.pop();
87  if (possibilities.empty()) {
88  // no more possibilities at this point, so undo our last move
89 
90  if (currentIdx == 0) {
91  // already at starting position, so no more permutations possible
92  break;
93  }
94 
95  currentIdx--;
96  seen[currentPermutation[currentIdx]] = 0;
97  currentPermutation.pop_back();
98 
99  // continue looking for permutations
100  continue;
101  }
102 
103  // try the next possibility
104  unsigned int nextNum = possibilities[0];
105 
106  // update the possibility vector for the current position
107  possibilities.erase(possibilities.begin());
108  todoStack.push(possibilities);
109 
110  if (seen[nextNum] != 0u) {
111  // number already seen in this permutation
112  continue;
113  } else {
114  // number can be used
115  seen[nextNum] = 1;
116  currentPermutation.push_back(nextNum);
117  currentIdx++;
118 
119  // if we havent reached the end of the permutation,
120  // push up the valid moves for the next position
121  if (currentIdx < clauseSize) {
122  todoStack.push(validMoves[currentIdx]);
123  }
124  }
125  }
126 
127  // found all permutations
128  return permutations;
129 }

131 /**
132  * Check if the atom at leftIdx in the left clause can potentially be matched up
133  * with the atom at rightIdx in the right clause.
134  * NOTE: index 0 refers to the head atom, index 1 to the first body atom, and so on.
135  */
136 bool isValidMove(const AstClause* left, size_t leftIdx, const AstClause* right, size_t rightIdx) {
137  // handle the case where one of the indices refers to the head
138  if (leftIdx == 0 && rightIdx == 0) {
139  const AstAtom* leftHead = left->getHead();
140  const AstAtom* rightHead = right->getHead();
141  return leftHead->getQualifiedName() == rightHead->getQualifiedName();
142  } else if (leftIdx == 0 || rightIdx == 0) {
143  return false;
144  }
145 
146  // both must hence be body atoms
147  int leftBodyAtomIdx = leftIdx - 1;
148  const AstAtom* leftAtom = dynamic_cast<AstAtom*>(left->getBodyLiterals()[leftBodyAtomIdx]);
149  assert(leftAtom != nullptr && "expected atom");
150 
151  int rightBodyAtomIdx = rightIdx - 1;
152  const AstAtom* rightAtom = dynamic_cast<AstAtom*>(right->getBodyLiterals()[rightBodyAtomIdx]);
153  assert(rightAtom != nullptr && "expected atom");
154 
155  return leftAtom->getQualifiedName() == rightAtom->getQualifiedName();
156 }

158 /**
159  * Check whether a valid variable mapping exists for the given permutation.
160  */
161 bool isValidPermutation(
162  const AstClause* left, const AstClause* right, const std::vector<unsigned int>& permutation) {
163  // --- perform the permutation ---
164 
165  auto reorderedLeft = std::unique_ptr<AstClause>(left->clone());
166 
167  // deduce the body atom permutation from the full clause permutation
168  std::vector<unsigned int> bodyPermutation(permutation.begin() + 1, permutation.end());
169  for (unsigned int& i : bodyPermutation) {
170  i -= 1;
171  }
172 
173  // currently, <permutation[i] == j> indicates that atom i should map to position j
174  // internally, for the clause class' reordering function, <permutation[i] == j> indicates
175  // that position i should contain atom j
176  // rearrange the permutation to match the internals
177  // TODO (azreika): perhaps change the internal reordering strategy because this came up in MST too
178  std::vector<unsigned int> reorderedPermutation(bodyPermutation.size());
179  for (size_t i = 0; i < bodyPermutation.size(); i++) {
180  reorderedPermutation[bodyPermutation[i]] = i;
181  }
182 
183  // perform the permutation
184  reorderedLeft.reset(reorderAtoms(reorderedLeft.get(), reorderedPermutation));
185 
186  // --- check if a valid variable exists corresponding to this permutation ---
187 
188  std::map<std::string, std::string> variableMap;
189  visitDepthFirst(*reorderedLeft, [&](const AstVariable& var) { variableMap[var.getName()] = ""; });
190 
191  // need to match the variables in the body
192  std::vector<AstLiteral*> leftAtoms = reorderedLeft->getBodyLiterals();
193  std::vector<AstLiteral*> rightAtoms = right->getBodyLiterals();
194 
195  // need to match the variables in the head
196  leftAtoms.push_back(reorderedLeft->getHead());
197  rightAtoms.push_back(right->getHead());
198 
199  // check if a valid variable mapping exists
200  auto isVariable = [](const AstArgument* arg) { return dynamic_cast<const AstVariable*>(arg) != nullptr; };
201 
202  bool validMapping = true;
203  for (size_t i = 0; i < leftAtoms.size() && validMapping; i++) {
204  // match arguments
205  assert(dynamic_cast<AstAtom*>(leftAtoms[i]) != nullptr && "expected atom");
206  assert(dynamic_cast<AstAtom*>(rightAtoms[i]) != nullptr && "expected atom");
207  std::vector<AstArgument*> leftArgs = dynamic_cast<AstAtom*>(leftAtoms[i])->getArguments();
208  std::vector<AstArgument*> rightArgs = dynamic_cast<AstAtom*>(rightAtoms[i])->getArguments();
209 
210  for (size_t j = 0; j < leftArgs.size(); j++) {
211  AstArgument* leftArg = leftArgs[j];
212  AstArgument* rightArg = rightArgs[j];
213  if (isVariable(leftArg) && isVariable(rightArg)) {
214  // both variables, their names should match to each other through the clause
215  std::string leftVarName = dynamic_cast<AstVariable*>(leftArg)->getName();
216  std::string rightVarName = dynamic_cast<AstVariable*>(rightArg)->getName();
217 
218  std::string currentMap = variableMap[leftVarName];
219  if (currentMap.empty()) {
220  // unassigned yet, so assign it appropriately
221  variableMap[leftVarName] = rightVarName;
222  } else if (currentMap != rightVarName) {
223  // mapping is inconsistent!
224  // clauses cannot be equivalent under this permutation
225  validMapping = false;
226  break;
227  }
228  } else if (castEq<AstStringConstant>(leftArg, rightArg)) {
229  validMapping = true;
230  } else if (castEq<AstNumericConstant>(leftArg, rightArg)) {
231  validMapping = true;
232  } else if (dynamic_cast<AstNilConstant*>(leftArg) != nullptr) {
233  validMapping = dynamic_cast<AstNilConstant*>(rightArg) != nullptr;
234  } else {
235  // not the same type, failed!
236  validMapping = false;
237  break;
238  }
239  }
240  }
241 
242  // return whether a valid variable mapping exists for this permutation
243  return validMapping;
244 }

246 /**
247  * Check whether two clauses are bijectively equivalent.
248  */
249 bool areBijectivelyEquivalent(const AstClause* left, const AstClause* right) {
250  // only check bijective equivalence for a subset of the possible clauses
251  auto isValidClause = [&](const AstClause* clause) {
252  // check that all body literals are atoms
253  // i.e. avoid clauses with constraints or negations
254  // TODO (azreika): extend to constraints and negations
255  for (AstLiteral* lit : clause->getBodyLiterals()) {
256  if (dynamic_cast<AstAtom*>(lit) == nullptr) {
257  return false;
258  }
259  }
260 
261  // check that all arguments are either constants or variables
262  // i.e. only allow primitive arguments
263  bool valid = true;
264  visitDepthFirst(*clause, [&](const AstArgument& arg) {
265  if (dynamic_cast<const AstVariable*>(&arg) == nullptr &&
266  dynamic_cast<const AstConstant*>(&arg) == nullptr) {
267  valid = false;
268  }
269  });
270  return valid;
271  };
272 
273  if (!isValidClause(left) || !isValidClause(right)) {
274  return false;
275  }
276 
277  // rules must be the same length to be equal
278  if (left->getBodyLiterals().size() != right->getBodyLiterals().size()) {
279  return false;
280  }
281 
282  // head atoms must have the same arity
283  if (left->getHead()->getArity() != right->getHead()->getArity()) {
284  return false;
285  }
286 
287  // set up the n x n permutation matrix, where n is the number of
288  // atoms in the clause, including the head atom
289  size_t size = left->getBodyLiterals().size() + 1;
290  auto permutationMatrix = std::vector<std::vector<unsigned int>>(size);
291  for (auto& i : permutationMatrix) {
292  i = std::vector<unsigned int>(size);
293  }
294 
295  // create permutation matrix
296  permutationMatrix[0][0] = 1;
297  for (size_t i = 1; i < size; i++) {
298  for (size_t j = 1; j < size; j++) {
299  if (isValidMove(left, i, right, j)) {
300  permutationMatrix[i][j] = 1;
301  }
302  }
303  }
304 
305  // check if any of these permutations have valid variable mappings associated with them
306  std::vector<std::vector<unsigned int>> permutations = extractPermutations(permutationMatrix);
307  for (auto permutation : permutations) {
308  if (isValidPermutation(left, right, permutation)) {
309  // valid permutation with valid corresponding variable mapping exists
310  // therefore, the two clauses are equivalent!
311  return true;
312  }
313  }
314 
315  return false;
316 }

318 /**
319  * Reduces locally-redundant clauses.
320  * A clause is locally-redundant if there is another clause within the same relation
321  * that computes the same set of tuples.
322  * @return true iff the program was changed
323  */
324 bool reduceLocallyEquivalentClauses(AstTranslationUnit& translationUnit) {
325  AstProgram& program = *translationUnit.getProgram();
326 
327  std::vector<AstClause*> clausesToDelete;
328 
329  // split up each relation's rules into equivalence classes
330  // TODO (azreika): consider turning this into an ast analysis instead
331  for (AstRelation* rel : program.getRelations()) {
332  std::vector<std::vector<AstClause*>> equivalenceClasses;
333 
334  for (AstClause* clause : getClauses(program, *rel)) {
335  bool added = false;
336 
337  for (std::vector<AstClause*>& eqClass : equivalenceClasses) {
338  AstClause* rep = eqClass[0];
339 
340  if (areBijectivelyEquivalent(rep, clause)) {
341  // clause belongs to an existing equivalence class, so delete it
342  eqClass.push_back(clause);
343  clausesToDelete.push_back(clause);
344  added = true;
345  break;
346  }
347  }
348 
349  if (!added) {
350  // clause does not belong to any existing equivalence class, so keep it
351  std::vector<AstClause*> clauseToAdd = {clause};
352  equivalenceClasses.push_back(clauseToAdd);
353  }
354  }
355  }
356 
357  // remove non-representative clauses
358  for (auto clause : clausesToDelete) {
359  program.removeClause(clause);
360  }
361 
362  // changed iff any clauses were deleted
363  return !clausesToDelete.empty();
364 }

366 /**
367  * Removes redundant singleton relations.
368  * Singleton relations are relations with a single clause. A singleton relation is redundant
369  * if there exists another singleton relation that computes the same set of tuples.
370  * @return true iff the program was changed
371  */
372 bool reduceSingletonRelations(AstTranslationUnit& translationUnit) {
373  // Note: This reduction is particularly useful in conjunction with the
374  // body-partitioning transformation
375  AstProgram& program = *translationUnit.getProgram();
376  auto* ioTypes = translationUnit.getAnalysis<IOType>();
377 
378  // Find all singleton relations to consider
379  std::vector<AstClause*> singletonRelationClauses;
380  for (AstRelation* rel : program.getRelations()) {
381  if (!ioTypes->isIO(rel) && getClauses(program, *rel).size() == 1) {
382  AstClause* clause = getClauses(program, *rel)[0];
383  singletonRelationClauses.push_back(clause);
384  }
385  }
386 
387  // Keep track of clauses found to be redundant
388  std::set<AstClause*> redundantClauses;
389 
390  // Keep track of canonical relation name for each redundant clause
391  std::map<AstQualifiedName, AstQualifiedName> canonicalName;
392 
393  // Check pairwise equivalence of each singleton relation
394  for (size_t i = 0; i < singletonRelationClauses.size(); i++) {
395  AstClause* first = singletonRelationClauses[i];
396  if (redundantClauses.find(first) != redundantClauses.end()) {
397  // Already found to be redundant, no need to check
398  continue;
399  }
400 
401  for (size_t j = i + 1; j < singletonRelationClauses.size(); j++) {
402  AstClause* second = singletonRelationClauses[j];
403 
404  // Note: Bijective-equivalence check does not care about the head relation name
405  if (areBijectivelyEquivalent(first, second)) {
406  AstQualifiedName firstName = first->getHead()->getQualifiedName();
407  AstQualifiedName secondName = second->getHead()->getQualifiedName();
408  redundantClauses.insert(second);
409  canonicalName.insert(std::pair(secondName, firstName));
410  }
411  }
412  }
413 
414  // Remove redundant relation definitions
415  for (AstClause* clause : redundantClauses) {
416  auto relName = clause->getHead()->getQualifiedName();
417  AstRelation* rel = getRelation(program, relName);
418  assert(rel != nullptr && "relation does not exist in program");
419  program.removeClause(clause);
420  program.removeRelation(relName);
421  }
422 
423  // Replace each redundant relation appearance with its canonical name
424  struct replaceRedundantRelations : public AstNodeMapper {
425  const std::map<AstQualifiedName, AstQualifiedName>& canonicalName;
426 
427  replaceRedundantRelations(const std::map<AstQualifiedName, AstQualifiedName>& canonicalName)
428  : canonicalName(canonicalName) {}
429 
430  std::unique_ptr<AstNode> operator()(std::unique_ptr<AstNode> node) const override {
431  // Remove appearances from children nodes
432  node->apply(*this);
433 
434  if (auto* atom = dynamic_cast<AstAtom*>(node.get())) {
435  auto pos = canonicalName.find(atom->getQualifiedName());
436  if (pos != canonicalName.end()) {
437  auto newAtom = std::unique_ptr<AstAtom>(atom->clone());
438  newAtom->setQualifiedName(pos->second);
439  return newAtom;
440  }
441  }
442 
443  return node;
444  }
445  };
446  replaceRedundantRelations update(canonicalName);
447  program.apply(update);
448 
449  // Program was changed iff a relation was replaced
450  return !canonicalName.empty();
451 }
