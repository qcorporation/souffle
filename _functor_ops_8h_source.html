95 /**
96  * Checks whether a functor operation can have a given argument count.
97  */
98 inline bool isValidFunctorOpArity(const FunctorOp op, const size_t arity) {
99  switch (op) {
100  /** Unary Functor Operators */
101  case FunctorOp::ORD:
102  case FunctorOp::STRLEN:
103  case FunctorOp::NEG:
104  case FunctorOp::FNEG:
105  case FunctorOp::BNOT:
106  case FunctorOp::UBNOT:
107  case FunctorOp::LNOT:
108  case FunctorOp::ULNOT:
109  case FunctorOp::TONUMBER:
110  case FunctorOp::TOSTRING:
111  case FunctorOp::ITOU:
112  case FunctorOp::UTOI:
113  case FunctorOp::ITOF:
114  case FunctorOp::FTOI:
115  case FunctorOp::UTOF:
116  case FunctorOp::FTOU:
117  return arity == 1;
118 
119  /** Binary Functor Operators */
120  case FunctorOp::ADD:
121  case FunctorOp::SUB:
122  case FunctorOp::MUL:
123  case FunctorOp::DIV:
124  case FunctorOp::EXP:
125  case FunctorOp::MOD:
126  case FunctorOp::BAND:
127  case FunctorOp::BOR:
128  case FunctorOp::BXOR:
129  case FunctorOp::BSHIFT_L:
130  case FunctorOp::BSHIFT_R:
131  case FunctorOp::BSHIFT_R_UNSIGNED:
132  case FunctorOp::LAND:
133  case FunctorOp::LOR:
134  case FunctorOp::UADD:
135  case FunctorOp::USUB:
136  case FunctorOp::UMUL:
137  case FunctorOp::UDIV:
138  case FunctorOp::UEXP:
139  case FunctorOp::UMOD:
140  case FunctorOp::UBAND:
141  case FunctorOp::UBOR:
142  case FunctorOp::UBXOR:
143  case FunctorOp::UBSHIFT_L:
144  case FunctorOp::UBSHIFT_R:
145  case FunctorOp::UBSHIFT_R_UNSIGNED:
146  case FunctorOp::ULAND:
147  case FunctorOp::ULOR:
148  case FunctorOp::FADD:
149  case FunctorOp::FSUB:
150  case FunctorOp::FMUL:
151  case FunctorOp::FDIV:
152  case FunctorOp::FEXP:
153  return arity == 2;
154 
155  /** Ternary Functor Operators */
156  case FunctorOp::SUBSTR:
157  return arity == 3;
158 
159  /** Non-fixed */
160  case FunctorOp::MAX:
161  case FunctorOp::MIN:
162  case FunctorOp::UMAX:
163  case FunctorOp::UMIN:
164  case FunctorOp::FMAX:
165  case FunctorOp::FMIN:
166  case FunctorOp::SMAX:
167  case FunctorOp::SMIN:
168  case FunctorOp::CAT:
169  return arity >= 2;
170  }
171 
172  assert(false && "unsupported operator");
173  return false;
174 }

359 /**
360  * Check the type of argument indicated by arg (0-indexed) of a functor op.
361  */
362 inline TypeAttribute functorOpArgType(const size_t arg, const FunctorOp op) {
363  switch (op) {
364  // Special case
365  case FunctorOp::ORD:
366  assert(false && "ord is a special function that returns a Ram Representation of the element");
367  case FunctorOp::ITOF:
368  case FunctorOp::ITOU:
369  case FunctorOp::NEG:
370  case FunctorOp::BNOT:
371  case FunctorOp::LNOT:
372  case FunctorOp::TOSTRING:
373  assert(arg == 0 && "unary functor out of bound");
374  return TypeAttribute::Signed;
375  case FunctorOp::FNEG:
376  case FunctorOp::FTOI:
377  case FunctorOp::FTOU:
378  assert(arg == 0 && "unary functor out of bound");
379  return TypeAttribute::Float;
380  case FunctorOp::STRLEN:
381  case FunctorOp::TONUMBER:
382  assert(arg == 0 && "unary functor out of bound");
383  return TypeAttribute::Symbol;
384  case FunctorOp::UBNOT:
385  case FunctorOp::ULNOT:
386  case FunctorOp::UTOI:
387  case FunctorOp::UTOF:
388  assert(arg == 0 && "unary functor out of bound");
389  return TypeAttribute::Unsigned;
390  case FunctorOp::ADD:
391  case FunctorOp::SUB:
392  case FunctorOp::MUL:
393  case FunctorOp::DIV:
394  case FunctorOp::EXP:
395  case FunctorOp::MOD:
396  case FunctorOp::BAND:
397  case FunctorOp::BOR:
398  case FunctorOp::BXOR:
399  case FunctorOp::BSHIFT_L:
400  case FunctorOp::BSHIFT_R:
401  case FunctorOp::BSHIFT_R_UNSIGNED:
402  case FunctorOp::LAND:
403  case FunctorOp::LOR:
404  assert(arg < 2 && "binary functor out of bound");
405  return TypeAttribute::Signed;
406  case FunctorOp::UADD:
407  case FunctorOp::USUB:
408  case FunctorOp::UMUL:
409  case FunctorOp::UDIV:
410  case FunctorOp::UEXP:
411  case FunctorOp::UMOD:
412  case FunctorOp::UBAND:
413  case FunctorOp::UBOR:
414  case FunctorOp::UBXOR:
415  case FunctorOp::UBSHIFT_L:
416  case FunctorOp::UBSHIFT_R:
417  case FunctorOp::UBSHIFT_R_UNSIGNED:
418  case FunctorOp::ULAND:
419  case FunctorOp::ULOR:
420  assert(arg < 2 && "binary functor out of bound");
421  return TypeAttribute::Unsigned;
422  case FunctorOp::FADD:
423  case FunctorOp::FSUB:
424  case FunctorOp::FMUL:
425  case FunctorOp::FDIV:
426  case FunctorOp::FEXP:
427  assert(arg < 2 && "binary functor out of bound");
428  return TypeAttribute::Float;
429  case FunctorOp::SUBSTR:
430  assert(arg < 3 && "ternary functor out of bound");
431  if (arg == 0) {
432  return TypeAttribute::Symbol;
433  } else {
434  return TypeAttribute::Signed; // In the future: Change to unsigned
435  }
436  case FunctorOp::MAX:
437  case FunctorOp::MIN:
438  return TypeAttribute::Signed;
439  case FunctorOp::UMAX:
440  case FunctorOp::UMIN:
441  return TypeAttribute::Unsigned;
442  case FunctorOp::FMAX:
443  case FunctorOp::FMIN:
444  return TypeAttribute::Float;
445  case FunctorOp::SMAX:
446  case FunctorOp::SMIN:
447  case FunctorOp::CAT:
448  return TypeAttribute::Symbol;
449  }
450  assert(false && "unsupported operator");
451  exit(EXIT_FAILURE);
452  return TypeAttribute::Record; // silence warning.
453 }

457  * At the moment, the signed versions are treated as representatives (because parser always returns a signed
458  * version).
459  */
460 inline bool isOverloadedFunctor(const FunctorOp functor) {
461  switch (functor) {
462  /* Unary */
463  case FunctorOp::NEG:
464  case FunctorOp::BNOT:
465  case FunctorOp::LNOT:
466  /* Binary */
467  case FunctorOp::ADD:
468  case FunctorOp::SUB:
469  case FunctorOp::MUL:
470  case FunctorOp::DIV:
471  case FunctorOp::EXP:
472  case FunctorOp::BAND:
473  case FunctorOp::BOR:
474  case FunctorOp::BXOR:
475  case FunctorOp::LAND:
476  case FunctorOp::LOR:
477  case FunctorOp::MOD:
478  case FunctorOp::MAX:
479  case FunctorOp::MIN:
480  return true;
481  break;
482  default:
483  break;
484  }

489 /**
490  * Convert an overloaded functor, so that it works with the requested type.
491  * Example: toType = Float, functor = PLUS -> FPLUS (version of plus working on floats)
492  */
493 inline FunctorOp convertOverloadedFunctor(const FunctorOp functor, const TypeAttribute toType) {
494  switch (functor) {
495  case FunctorOp::NEG:
496  assert(toType == TypeAttribute::Float && "Invalid functor conversion from NEG");
497  return FunctorOp::FNEG;
498  case FunctorOp::BNOT:
499  assert(toType == TypeAttribute::Unsigned && "Invalid functor conversion from UBNOT");
500  return FunctorOp::UBNOT;
501  case FunctorOp::LNOT:
502  assert(toType == TypeAttribute::Unsigned && "Invalid functor conversion from LNOT");
503  return FunctorOp::ULNOT;
504  /* Binary */
505  case FunctorOp::ADD:
506  if (toType == TypeAttribute::Float) {
507  return FunctorOp::FADD;
508  } else if (toType == TypeAttribute::Unsigned) {
509  return FunctorOp::UADD;
510  }
511  assert(false && "Invalid functor conversion");
512  break;
513  case FunctorOp::SUB:
514  if (toType == TypeAttribute::Float) {
515  return FunctorOp::FSUB;
516  } else if (toType == TypeAttribute::Unsigned) {
517  return FunctorOp::USUB;
518  }
519  assert(false && "Invalid functor conversion");
520  break;
521  case FunctorOp::MUL:
522  if (toType == TypeAttribute::Float) {
523  return FunctorOp::FMUL;
524  } else if (toType == TypeAttribute::Unsigned) {
525  return FunctorOp::UMUL;
526  }
527  assert(false && "Invalid functor conversion");
528  break;
529  case FunctorOp::DIV:
530  if (toType == TypeAttribute::Float) {
531  return FunctorOp::FDIV;
532  } else if (toType == TypeAttribute::Unsigned) {
533  return FunctorOp::UDIV;
534  }
535  assert(false && "Invalid functor conversion");
536  break;
537  case FunctorOp::EXP:
538  if (toType == TypeAttribute::Float) {
539  return FunctorOp::FEXP;
540  } else if (toType == TypeAttribute::Unsigned) {
541  return FunctorOp::UEXP;
542  }
543  assert(false && "Invalid functor conversion");
544  break;
545  case FunctorOp::MAX:
546  if (toType == TypeAttribute::Float) {
547  return FunctorOp::FMAX;
548  } else if (toType == TypeAttribute::Unsigned) {
549  return FunctorOp::UMAX;
550  } else if (toType == TypeAttribute::Symbol) {
551  return FunctorOp::SMAX;
552  }
553  assert(false && "Invalid functor conversion");
554  break;
555  case FunctorOp::MIN:
556  if (toType == TypeAttribute::Float) {
557  return FunctorOp::FMIN;
558  } else if (toType == TypeAttribute::Unsigned) {
559  return FunctorOp::UMIN;
560  } else if (toType == TypeAttribute::Symbol) {
561  return FunctorOp::SMIN;
562  }
563  assert(false && "Invalid functor conversion");
564  break;
565  case FunctorOp::BAND:
566  assert(toType == TypeAttribute::Unsigned && "Invalid functor conversion");
567  return FunctorOp::UBAND;
568  case FunctorOp::BOR:
569  assert(toType == TypeAttribute::Unsigned && "Invalid functor conversion");
570  return FunctorOp::UBOR;
571  case FunctorOp::BXOR:
572  assert(toType == TypeAttribute::Unsigned && "Invalid functor conversion");
573  return FunctorOp::UBXOR;
574  case FunctorOp::BSHIFT_L:
575  assert(toType == TypeAttribute::Unsigned && "Invalid functor conversion");
576  return FunctorOp::UBSHIFT_L;
577  case FunctorOp::BSHIFT_R:
578  assert(toType == TypeAttribute::Unsigned && "Invalid functor conversion");
579  return FunctorOp::UBSHIFT_R;
580  case FunctorOp::LAND:
581  assert(toType == TypeAttribute::Unsigned && "Invalid functor conversion");
582  return FunctorOp::ULAND;
583  case FunctorOp::LOR:
584  assert(toType == TypeAttribute::Unsigned && "Invalid functor conversion");
585  return FunctorOp::ULOR;
586  case FunctorOp::MOD:
587  assert(toType == TypeAttribute::Unsigned && "Invalid functor conversion");
588  return FunctorOp::UMOD;
589  default:
590  assert(false && "Invalid functor");
591  }
592 }
