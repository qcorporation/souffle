97  void outputRecord(std::ostream& destination, const RamDomain value, const std::string& name) {
98  Json recordInfo = types["records"][name];
99 
100  // Check if record type information are present
101  assert(!recordInfo.is_null() && "Missing record type information");
102 
103  // Check for nil

106  return;
107  }
108 
109  const Json recordTypes = recordInfo["types"];
110  const size_t recordArity = recordInfo["arity"].long_value();
111 
112  const RamDomain* tuplePtr = recordTable.unpack(value, recordArity);
113 
114  destination << "[";
115 
116  // print record's elements
117  for (size_t i = 0; i < recordArity; ++i) {
118  if (i > 0) {
119  destination << ", ";
120  }
121 
122  const std::string& recordType = recordTypes[i].string_value();
123  const RamDomain recordValue = tuplePtr[i];
124 
125  switch (recordType[0]) {
126  case 'i':
127  destination << recordValue;
128  break;
129  case 'f':
130  destination << ramBitCast<RamFloat>(recordValue);
131  break;
132  case 'u':
133  destination << ramBitCast<RamUnsigned>(recordValue);
134  break;
135  case 's':
136  destination << symbolTable.unsafeResolve(recordValue);
137  break;
138  case 'r':
139  outputRecord(destination, recordValue, recordType);
140  break;
141  default:

144  }
