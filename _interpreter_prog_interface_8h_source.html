117 
118  /** Get current tuple */
119  tuple& operator*() override {
120  // set tuple stream to first element
121  tup.rewind();
122 
123  // construct the tuple to return
124  for (size_t i = 0; i < ramRelationInterface->getArity(); i++) {
125  switch (*(ramRelationInterface->getAttrType(i))) {
126  case 's': {
127  std::string s = ramRelationInterface->getSymbolTable().resolve((*it)[i]);
128  tup << s;
129  break;
130  }
131  case 'f':
132  tup << ramBitCast<RamFloat>((*it)[i]);
133  break;
134  case 'u':
135  tup << ramBitCast<RamUnsigned>((*it)[i]);
136  break;
137  default:
138  tup << (*it)[i];
139  break;
140  }
141  }
142  tup.rewind();
143  return tup;
144  }

188 /**
189  * Implementation of SouffleProgram interface for an interpreter instance
190  */
191 class InterpreterProgInterface : public SouffleProgram {
192 public:
193  explicit InterpreterProgInterface(InterpreterEngine& interp)
194  : prog(interp.getTranslationUnit().getProgram()), exec(interp),
195  symTable(interp.getTranslationUnit().getSymbolTable()) {
196  uint32_t id = 0;
197 
198  // Retrieve AST Relations and store them in a map
199  std::map<std::string, const RamRelation*> map;
200  visitDepthFirst(prog, [&](const RamRelation& rel) { map[rel.getName()] = &rel; });
201 
202  // Build wrapper relations for Souffle's interface
203  for (auto& relHandler : exec.getRelationMap()) {
204  if (relHandler == nullptr) {
205  // Skip droped relation.
206  continue;
207  }
208  auto& interpreterRel = **relHandler;
209  auto& name = interpreterRel.getName();
210  assert(map[name]);
211  const RamRelation& rel = *map[name];
212 
213  // construct types and names vectors
214  std::vector<std::string> types = rel.getAttributeTypes();
215  std::vector<std::string> attrNames = rel.getAttributeNames();
216 
217  auto* interface = new InterpreterRelInterface(
218  interpreterRel, symTable, rel.getName(), types, attrNames, id);
219  interfaces.push_back(interface);
220  bool input = false;
221  bool output = false;
222  visitDepthFirst(prog, [&](const RamIO& io) {
223  if (io.getRelation() == rel) {
224  const std::string& op = io.get("operation");
225  if (op == "input") {
226  input = true;
227  } else if (op == "output" || op == "printsize") {
228  output = true;
229  } else {
230  assert("wrong i/o operation");
231  }
232  }
233  });
234  addRelation(rel.getName(), interface, input, output);
235  id++;
236  }
237  }
238  ~InterpreterProgInterface() override {
239  for (auto* interface : interfaces) {
240  delete interface;
241  }
242  }
243 
244  /** Run program instance: not implemented */
245  void run() override {}
246 
247  /** Load data, run program instance, store data: not implemented */
248  void runAll(std::string, std::string) override {}

250  /** Load input data: not implemented */
251  void loadAll(std::string) override {}

254  void printAll(std::string) override {}
