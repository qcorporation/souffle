50 bool Order::valid() const {
51  // Check that all indices are in range.
52  for (int i : order) {
53  if (i < 0 || i >= int(order.size())) {
54  return false;
55  }
56  }
57  // Check that there are no duplicates.
58  for (std::size_t i = 0; i < order.size(); i++) {
59  for (std::size_t j = i + 1; j < order.size(); j++) {
60  if (order[i] == order[j]) {
61  return false;
62  }
63  }
64  }
65  return true;
66 }

215 class GenericIndex : public InterpreterIndex {
216 protected:
217  using Entry = typename Structure::element_type;
218  using Hints = typename Structure::operation_hints;

229  // a source adapter for streaming through data
230  class Source : public Stream::Source {
231  const Order& order;
232 
233  // the begin and end of the stream
234  iter cur;
235  iter end;
236 
237  // an internal buffer for re-ordered elements
238  std::array<Entry, Stream::BUFFER_SIZE> buffer;
239 
240  public:
241  Source(const Order& order, iter begin, iter end)
242  : order(order), cur(std::move(begin)), end(std::move(end)) {}
243 
244  int load(TupleRef* out, int max) override {
245  int c = 0;
246  while (cur != end && c < max) {
247  buffer[c] = order.decode(*cur);
248  out[c] = buffer[c];
249  ++cur;
250  ++c;
251  }
252  return c;
253  }
254 
255  int reload(TupleRef* out, int max) override {
256  int c = 0;
257  max = std::min(max, Stream::BUFFER_SIZE);
258  while (c < max) {
259  out[c] = buffer[c];
260  ++c;
261  }
262  return c;
263  }
264 
265  std::unique_ptr<Stream::Source> clone() override {
266  auto source = std::make_unique<Source>(order, cur, end);
267  source->buffer = this->buffer;
268  return source;
269  }
270  };

379  /* lexicographical comparison operation on two tuple pointers */
380  struct comparator {
381  const std::vector<int> order;
382 
383  /* constructor to initialize state */
384  comparator(std::vector<int> order) : order(std::move(order)) {}
385 
386  /* comparison function */
387  int operator()(const TupleRef& x, const TupleRef& y) const {
388  for (auto& i : order) {
389  if (x[i] < y[i]) {
390  return -1;
391  }
392  if (x[i] > y[i]) {
393  return 1;
394  }
395  }
396  return 0;
397  }
398 
399  /* less comparison */
400  bool less(const TupleRef& x, const TupleRef& y) const {
401  return operator()(x, y) < 0;
402  }
403 
404  /* equal comparison */
405  bool equal(const TupleRef& x, const TupleRef& y) const {
406  for (auto& i : order) {
407  if (x[i] != y[i]) {
408  return false;
409  }
410  }
411  return true;
412  }
413  };
