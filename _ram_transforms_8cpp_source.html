37 bool ExpandFilterTransformer::expandFilters(RamProgram& program) {
38  bool changed = false;
39  visitDepthFirst(program, [&](const RamQuery& query) {
40  std::function<std::unique_ptr<RamNode>(std::unique_ptr<RamNode>)> filterRewriter =
41  [&](std::unique_ptr<RamNode> node) -> std::unique_ptr<RamNode> {
42  if (const RamFilter* filter = dynamic_cast<RamFilter*>(node.get())) {
43  const RamCondition* condition = &filter->getCondition();
44  std::vector<std::unique_ptr<RamCondition>> conditionList = toConjunctionList(condition);
45  if (conditionList.size() > 1) {
46  changed = true;
47  std::vector<std::unique_ptr<RamFilter>> filters;
48  for (auto& cond : conditionList) {
49  auto tempCond = cond->clone();
50  if (filters.empty()) {
51  filters.emplace_back(
52  std::make_unique<RamFilter>(std::unique_ptr<RamCondition>(tempCond),
53  std::unique_ptr<RamOperation>(filter->getOperation().clone())));
54  } else {
55  filters.emplace_back(std::make_unique<RamFilter>(
56  std::unique_ptr<RamCondition>(tempCond), std::move(filters.back())));
57  }
58  }
59  node = std::move(filters.back());
60  }
61  }
62  node->apply(makeLambdaRamMapper(filterRewriter));
63  return node;
64  };
65  const_cast<RamQuery*>(&query)->apply(makeLambdaRamMapper(filterRewriter));
66  });
67  return changed;
68 }

70 bool ReorderConditionsTransformer::reorderConditions(RamProgram& program) {
71  bool changed = false;
72  visitDepthFirst(program, [&](const RamQuery& query) {
73  std::function<std::unique_ptr<RamNode>(std::unique_ptr<RamNode>)> filterRewriter =
74  [&](std::unique_ptr<RamNode> node) -> std::unique_ptr<RamNode> {
75  if (const RamFilter* filter = dynamic_cast<RamFilter*>(node.get())) {
76  const RamCondition* condition = &filter->getCondition();
77  std::vector<std::unique_ptr<RamCondition>> sortedConds;
78  std::vector<std::unique_ptr<RamCondition>> condList = toConjunctionList(condition);
79  for (auto& cond : condList) {
80  sortedConds.emplace_back(cond->clone());
81  }
82  std::sort(sortedConds.begin(), sortedConds.end(),
83  [&](std::unique_ptr<RamCondition>& a, std::unique_ptr<RamCondition>& b) {
84  return rca->getComplexity(a.get()) < rca->getComplexity(b.get());
85  });
86 
87  if (!std::equal(sortedConds.begin(), sortedConds.end(), condList.begin(),
88  [](std::unique_ptr<RamCondition>& a, std::unique_ptr<RamCondition>& b) {
89  return *a == *b;
90  })) {
91  changed = true;
92  node = std::make_unique<RamFilter>(
93  std::unique_ptr<RamCondition>(toCondition(sortedConds)),
94  std::unique_ptr<RamOperation>(filter->getOperation().clone()));
95  }
96  }
97  node->apply(makeLambdaRamMapper(filterRewriter));
98  return node;
99  };
100  const_cast<RamQuery*>(&query)->apply(makeLambdaRamMapper(filterRewriter));
101  });
102  return changed;
103 }

105 bool CollapseFiltersTransformer::collapseFilters(RamProgram& program) {
106  bool changed = false;
107  visitDepthFirst(program, [&](const RamQuery& query) {
108  std::function<std::unique_ptr<RamNode>(std::unique_ptr<RamNode>)> filterRewriter =
109  [&](std::unique_ptr<RamNode> node) -> std::unique_ptr<RamNode> {
110  if (const RamFilter* filter = dynamic_cast<RamFilter*>(node.get())) {
111  // true if two consecutive filters in loop nest found
112  bool canCollapse = false;
113 
114  // storing conditions for collapsing
115  std::vector<std::unique_ptr<RamCondition>> conditions;
116 
117  const RamFilter* prevFilter = filter;
118  conditions.emplace_back(filter->getCondition().clone());
119  while (auto* nextFilter = dynamic_cast<RamFilter*>(&prevFilter->getOperation())) {
120  canCollapse = true;
121  conditions.emplace_back(nextFilter->getCondition().clone());
122  prevFilter = nextFilter;
123  }
124 
125  if (canCollapse) {
126  changed = true;
127  node = std::make_unique<RamFilter>(toCondition(conditions),
128  std::unique_ptr<RamOperation>(
129  dynamic_cast<RamOperation*>(prevFilter->getOperation().clone())),
130  prevFilter->getProfileText());
131  }
132  }
133  node->apply(makeLambdaRamMapper(filterRewriter));
134  return node;
135  };
136  const_cast<RamQuery*>(&query)->apply(makeLambdaRamMapper(filterRewriter));
137  });

141 bool EliminateDuplicatesTransformer::eliminateDuplicates(RamProgram& program) {
142  bool changed = false;
143  visitDepthFirst(program, [&](const RamQuery& query) {
144  std::function<std::unique_ptr<RamNode>(std::unique_ptr<RamNode>)> filterRewriter =
145  [&](std::unique_ptr<RamNode> node) -> std::unique_ptr<RamNode> {
146  if (const RamFilter* filter = dynamic_cast<RamFilter*>(node.get())) {
147  const RamCondition* condition = &filter->getCondition();
148  std::vector<std::unique_ptr<RamCondition>> conds = toConjunctionList(condition);
149  bool eliminatedDuplicate = false;
150  for (std::size_t i = 0; i < conds.size(); i++) {
151  for (std::size_t j = i + 1; j < conds.size(); j++) {
152  if (*conds[i] == *conds[j]) {
153  conds.erase(conds.begin() + j);
154  i = -1;
155  eliminatedDuplicate = true;
156  break;
157  }
158  }
159  }
160  if (eliminatedDuplicate) {
161  changed = true;
162  node = std::make_unique<RamFilter>(std::unique_ptr<RamCondition>(toCondition(conds)),
163  std::unique_ptr<RamOperation>(filter->getOperation().clone()));
164  }
165  }
166  node->apply(makeLambdaRamMapper(filterRewriter));
167  return node;
168  };
169  const_cast<RamQuery*>(&query)->apply(makeLambdaRamMapper(filterRewriter));
170  });
171  return changed;

173 bool HoistConditionsTransformer::hoistConditions(RamProgram& program) {
174  bool changed = false;
175 
176  // helper for collecting conditions from filter operations
177  auto addCondition = [](std::unique_ptr<RamCondition> condition,
178  RamCondition* c) -> std::unique_ptr<RamCondition> {
179  if (condition == nullptr) {
180  return std::unique_ptr<RamCondition>(c);
181  } else {
182  return std::make_unique<RamConjunction>(std::move(condition), std::unique_ptr<RamCondition>(c));
183  }
184  };
185 
186  // hoist conditions to the most outer scope if they
187  // don't depend on RamTupleOperations
188  visitDepthFirst(program, [&](const RamQuery& query) {
189  std::unique_ptr<RamCondition> newCondition;
190  std::function<std::unique_ptr<RamNode>(std::unique_ptr<RamNode>)> filterRewriter =
191  [&](std::unique_ptr<RamNode> node) -> std::unique_ptr<RamNode> {
192  if (auto* filter = dynamic_cast<RamFilter*>(node.get())) {
193  const RamCondition& condition = filter->getCondition();
194  // if filter condition is independent of any RamTupleOperation,
195  // delete the filter operation and collect condition
196  if (rla->getLevel(&condition) == -1) {
197  changed = true;
198  newCondition = addCondition(std::move(newCondition), condition.clone());
199  node->apply(makeLambdaRamMapper(filterRewriter));
200  return std::unique_ptr<RamOperation>(filter->getOperation().clone());
201  }
202  }
203  node->apply(makeLambdaRamMapper(filterRewriter));
204  return node;
205  };
206  auto* mQuery = const_cast<RamQuery*>(&query);
207  mQuery->apply(makeLambdaRamMapper(filterRewriter));
208  if (newCondition != nullptr) {
209  // insert new filter operation at outer-most level of the query
210  changed = true;
211  auto* nestedOp = const_cast<RamOperation*>(&mQuery->getOperation());
212  mQuery->rewrite(nestedOp, std::make_unique<RamFilter>(std::move(newCondition),
213  std::unique_ptr<RamOperation>(nestedOp->clone())));
214  }
215  });
216 
217  // hoist conditions for each RamTupleOperation operation
218  visitDepthFirst(program, [&](const RamTupleOperation& search) {
219  std::unique_ptr<RamCondition> newCondition;
220  std::function<std::unique_ptr<RamNode>(std::unique_ptr<RamNode>)> filterRewriter =
221  [&](std::unique_ptr<RamNode> node) -> std::unique_ptr<RamNode> {
222  if (auto* filter = dynamic_cast<RamFilter*>(node.get())) {
223  const RamCondition& condition = filter->getCondition();
224  // if filter condition matches level of RamTupleOperation,
225  // delete the filter operation and collect condition
226  if (rla->getLevel(&condition) == search.getTupleId()) {
227  changed = true;
228  newCondition = addCondition(std::move(newCondition), condition.clone());
229  node->apply(makeLambdaRamMapper(filterRewriter));
230  return std::unique_ptr<RamOperation>(filter->getOperation().clone());
231  }
232  }
233  node->apply(makeLambdaRamMapper(filterRewriter));
234  return node;
235  };
236  auto* tupleOp = const_cast<RamTupleOperation*>(&search);
237  tupleOp->apply(makeLambdaRamMapper(filterRewriter));
238  if (newCondition != nullptr) {
239  // insert new filter operation after the search operation
240  changed = true;
241  tupleOp->rewrite(&tupleOp->getOperation(),
242  std::make_unique<RamFilter>(std::move(newCondition),
243  std::unique_ptr<RamOperation>(tupleOp->getOperation().clone())));
244  }
245  });
246  return changed;
247 }

249 std::unique_ptr<RamExpression> MakeIndexTransformer::getExpression(
250  RamCondition* c, size_t& element, int identifier) {
251  if (auto* binRelOp = dynamic_cast<RamConstraint*>(c)) {
252  // TODO: FIXME: how does this interact w/ `FEQ`?
253  if (isEqConstraint(binRelOp->getOperator())) {
254  if (const auto* lhs = dynamic_cast<const RamTupleElement*>(&binRelOp->getLHS())) {
255  const RamExpression* rhs = &binRelOp->getRHS();
256  if (lhs->getTupleId() == identifier && rla->getLevel(rhs) < identifier) {
257  element = lhs->getElement();
258  return std::unique_ptr<RamExpression>(rhs->clone());
259  }
260  }
261  if (const auto* rhs = dynamic_cast<const RamTupleElement*>(&binRelOp->getRHS())) {
262  const RamExpression* lhs = &binRelOp->getLHS();
263  if (rhs->getTupleId() == identifier && rla->getLevel(lhs) < identifier) {
264  element = rhs->getElement();
265  return std::unique_ptr<RamExpression>(lhs->clone());
266  }
267  }
268  }
269  }
270  return nullptr;
271 }

273 std::unique_ptr<RamCondition> MakeIndexTransformer::constructPattern(
274  std::vector<std::unique_ptr<RamExpression>>& queryPattern, bool& indexable,
275  std::vector<std::unique_ptr<RamCondition>> conditionList, int identifier) {
276  // Remaining conditions which cannot be handled by an index
277  std::unique_ptr<RamCondition> condition;
278  auto addCondition = [&](std::unique_ptr<RamCondition> c) {
279  if (condition != nullptr) {
280  condition = std::make_unique<RamConjunction>(std::move(condition), std::move(c));
281  } else {
282  condition = std::move(c);
283  }
284  };
285 
286  // Build query pattern and remaining condition
287  for (auto& cond : conditionList) {
288  size_t element = 0;
289  if (std::unique_ptr<RamExpression> value = getExpression(cond.get(), element, identifier)) {
290  if (queryPattern[element] == nullptr) {
291  indexable = true;
292  queryPattern[element] = std::move(value);
293  } else {
294  // FIXME: `FEQ` handling; need to know if the expr is a float exp or not
295  addCondition(std::make_unique<RamConstraint>(BinaryConstraintOp::EQ, std::move(value),
296  std::unique_ptr<RamExpression>(queryPattern[element]->clone())));
297  }
298  } else {
299  addCondition(std::move(cond));
300  }
301  }
302 
303  // Avoid null-pointers for condition and query pattern
304  if (condition == nullptr) {
305  condition = std::make_unique<RamTrue>();
306  }
307 
308  for (auto& p : queryPattern) {
309  if (p == nullptr) {
310  p = std::make_unique<RamUndefValue>();
311  }
312  }
313  return condition;
314 }

355 std::unique_ptr<RamOperation> MakeIndexTransformer::rewriteIndexScan(const RamIndexScan* iscan) {
356  if (const auto* filter = dynamic_cast<const RamFilter*>(&iscan->getOperation())) {
357  const RamRelation& rel = iscan->getRelation();
358  const int identifier = iscan->getTupleId();
359  std::vector<std::unique_ptr<RamExpression>> queryPattern(rel.getArity());
360  bool indexable = false;
361  std::unique_ptr<RamCondition> condition = constructPattern(
362  queryPattern, indexable, toConjunctionList(&filter->getCondition()), identifier);
363  if (indexable) {
364  // Merge Index Pattern here
365  const auto prevPattern = iscan->getRangePattern();
366  auto addCondition = [&](std::unique_ptr<RamCondition> c) {
367  if (condition != nullptr) {
368  condition = std::make_unique<RamConjunction>(std::move(condition), std::move(c));
369  } else {
370  condition = std::move(c);
371  }
372  };
373  for (std::size_t i = 0; i < rel.getArity(); i++) {
374  if (prevPattern[i] != nullptr) {
375  if (queryPattern[i] == nullptr) {
376  // found a new indexable attribute
377  queryPattern[i] = std::unique_ptr<RamExpression>(prevPattern[i]->clone());
378  } else {
379  // found a new constraint that is not dependent on the current scan level
380  // and can be hoisted in a later transformation.
381  // FIXME: `FEQ` handling; need to know if the expr is a float exp or not
382  addCondition(std::make_unique<RamConstraint>(BinaryConstraintOp::EQ,
383  std::unique_ptr<RamExpression>(prevPattern[i]->clone()),
384  std::unique_ptr<RamExpression>(queryPattern[i]->clone())));
385  }
386  }
387  }
388  std::unique_ptr<RamOperation> op = std::unique_ptr<RamOperation>(filter->getOperation().clone());
389  if (!isRamTrue(condition.get())) {
390  op = std::make_unique<RamFilter>(std::move(condition), std::move(op));
391  }
392  return std::make_unique<RamIndexScan>(std::make_unique<RamRelationReference>(&rel), identifier,
393  std::move(queryPattern), std::move(op), iscan->getProfileText());
394  }
395  }
396  return nullptr;
397 }

455 std::unique_ptr<RamOperation> IfConversionTransformer::rewriteIndexScan(const RamIndexScan* indexScan) {
456  // check whether tuple is used in subsequent operations
457  bool tupleNotUsed = true;
458  visitDepthFirst(*indexScan, [&](const RamTupleElement& element) {
459  if (element.getTupleId() == indexScan->getTupleId()) {
460  tupleNotUsed = false;

462  });
463 
464  // if not used, transform the IndexScan operation to an existence check
465  if (tupleNotUsed) {
466  // replace IndexScan with an Filter/Existence check
467  std::vector<std::unique_ptr<RamExpression>> newValues;
468  for (auto& cur : indexScan->getRangePattern()) {
469  RamExpression* val = nullptr;
470  if (cur != nullptr) {
471  val = cur->clone();
472  }
473  newValues.emplace_back(val);
474  }
475 
476  // check if there is a break statement nested in the Scan - if so, remove it
477  RamOperation* newOp;
478  if (const auto* breakOp = dynamic_cast<const RamBreak*>(&indexScan->getOperation())) {
479  newOp = breakOp->getOperation().clone();
480  } else {

483 
484  return std::make_unique<RamFilter>(

486  std::make_unique<RamRelationReference>(&indexScan->getRelation()),

636  }
637  }
638  node->apply(makeLambdaRamMapper(elementRewriter));

640  };

647 bool HoistAggregateTransformer::hoistAggregate(RamProgram& program) {
648  bool changed = false;
649 
650  // There are two cases: aggregates that have no data-dependencies on
651  // other RAM operations, and aggregates that have data-dependencies.
652  // A rewriter has two tasks: (1) identify a single aggregate that
653  // can be hoisted and (2) insert it at the outermost level.
654  // We assume all RamOperations are renumbered for this transformation.
655 
656  // Hoist a single aggregate to an outer scope that is data-independent.
657  visitDepthFirst(program, [&](const RamQuery& query) {
658  std::unique_ptr<RamNestedOperation> newAgg;
659  bool priorTupleOp = false;
660  std::function<std::unique_ptr<RamNode>(std::unique_ptr<RamNode>)> aggRewriter =
661  [&](std::unique_ptr<RamNode> node) -> std::unique_ptr<RamNode> {
662  if (nullptr != dynamic_cast<RamAggregate*>(node.get())) {
663  auto* tupleOp = dynamic_cast<RamTupleOperation*>(node.get());
664  assert(tupleOp != nullptr && "aggregate conversion to tuple operation failed");
665  if (rla->getLevel(tupleOp) == -1 && !priorTupleOp) {
666  changed = true;
667  newAgg = std::unique_ptr<RamNestedOperation>(
668  dynamic_cast<RamNestedOperation*>(tupleOp->clone()));
669  assert(newAgg != nullptr && "failed to make a clone");
670  return std::unique_ptr<RamOperation>(tupleOp->getOperation().clone());
671  }
672  } else if (nullptr != dynamic_cast<RamTupleOperation*>(node.get())) {
673  // tuple operation that is a non-aggregate
674  priorTupleOp = true;
675  }
676  node->apply(makeLambdaRamMapper(aggRewriter));
677  return node;
678  };
679  const_cast<RamQuery*>(&query)->apply(makeLambdaRamMapper(aggRewriter));
680  if (newAgg != nullptr) {
681  newAgg->rewrite(
682  &newAgg->getOperation(), std::unique_ptr<RamOperation>(query.getOperation().clone()));
683  const_cast<RamQuery*>(&query)->rewrite(&query.getOperation(), std::move(newAgg));
684  }
685  });
686 
687  // hoist a single aggregate to an outer scope that is data-dependent on a prior operation.
688  visitDepthFirst(program, [&](const RamQuery& query) {
689  int newLevel = -1;
690  std::unique_ptr<RamNestedOperation> newAgg;
691  int priorOpLevel = -1;
692 
693  std::function<std::unique_ptr<RamNode>(std::unique_ptr<RamNode>)> aggRewriter =
694  [&](std::unique_ptr<RamNode> node) -> std::unique_ptr<RamNode> {
695  if (nullptr != dynamic_cast<RamAbstractAggregate*>(node.get())) {
696  auto* tupleOp = dynamic_cast<RamTupleOperation*>(node.get());
697  assert(tupleOp != nullptr && "aggregate conversion to nested operation failed");
698  int dataDepLevel = rla->getLevel(tupleOp);
699  if (dataDepLevel != -1 && dataDepLevel < tupleOp->getTupleId() - 1) {
700  // If all tuple ops between the data-dependence level and agg
701  // are aggregates, then we do not hoist, i.e., we would
702  // continuously swap their positions.
703  if (dataDepLevel != priorOpLevel) {
704  changed = true;
705  newLevel = dataDepLevel;
706  newAgg = std::unique_ptr<RamNestedOperation>(
707  dynamic_cast<RamNestedOperation*>(tupleOp->clone()));
708  assert(newAgg != nullptr && "failed to make a clone");
709  return std::unique_ptr<RamOperation>(tupleOp->getOperation().clone());
710  }
711  }
712  } else if (const RamTupleOperation* tupleOp = dynamic_cast<RamTupleOperation*>(node.get())) {
713  priorOpLevel = tupleOp->getTupleId();
714  }
715  node->apply(makeLambdaRamMapper(aggRewriter));
716  if (auto* search = dynamic_cast<RamTupleOperation*>(node.get())) {
717  if (newAgg != nullptr && search->getTupleId() == newLevel) {
718  newAgg->rewrite(&newAgg->getOperation(),
719  std::unique_ptr<RamOperation>(search->getOperation().clone()));
720  search->rewrite(&search->getOperation(), std::move(newAgg));
721  }
722  }
723  return node;
724  };
725  const_cast<RamQuery*>(&query)->apply(makeLambdaRamMapper(aggRewriter));
726  });
727  return changed;
728 } // namespace souffle

730 bool ParallelTransformer::parallelizeOperations(RamProgram& program) {
731  bool changed = false;
732 
733  // parallelize the most outer loop only
734  // most outer loops can be scan/choice/indexScan/indexChoice
735  visitDepthFirst(program, [&](const RamQuery& query) {
736  std::function<std::unique_ptr<RamNode>(std::unique_ptr<RamNode>)> parallelRewriter =
737  [&](std::unique_ptr<RamNode> node) -> std::unique_ptr<RamNode> {
738  if (const RamScan* scan = dynamic_cast<RamScan*>(node.get())) {
739  if (scan->getTupleId() == 0 && scan->getRelation().getArity() > 0) {
740  if (nullptr == dynamic_cast<RamProject*>(&scan->getOperation())) {
741  changed = true;
742  return std::make_unique<RamParallelScan>(
743  std::make_unique<RamRelationReference>(&scan->getRelation()),
744  scan->getTupleId(),
745  std::unique_ptr<RamOperation>(scan->getOperation().clone()),
746  scan->getProfileText());
747  }
748  }
749  } else if (const RamChoice* choice = dynamic_cast<RamChoice*>(node.get())) {
750  if (choice->getTupleId() == 0) {
751  changed = true;
752  return std::make_unique<RamParallelChoice>(
753  std::make_unique<RamRelationReference>(&choice->getRelation()),
754  choice->getTupleId(),
755  std::unique_ptr<RamCondition>(choice->getCondition().clone()),
756  std::unique_ptr<RamOperation>(choice->getOperation().clone()),
757  choice->getProfileText());
758  }
759  } else if (const RamIndexScan* indexScan = dynamic_cast<RamIndexScan*>(node.get())) {
760  if (indexScan->getTupleId() == 0) {
761  changed = true;
762  const RamRelation& rel = indexScan->getRelation();
763  std::vector<std::unique_ptr<RamExpression>> queryPattern;
764  for (const RamExpression* cur : indexScan->getRangePattern()) {

778  if (indexChoice->getTupleId() == 0) {

785  } else {

787  }
