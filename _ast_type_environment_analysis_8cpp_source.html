36 /**
37  * A utility function utilized by the finishParsing member function to update a type environment
38  * out of a given list of types in the AST
39  *
40  * @param types the types specified in the input file, contained in the AST
41  * @param env the type environment to be updated
42  */
43 void TypeEnvironmentAnalysis::updateTypeEnvironment(const AstProgram& program) {
44  // build up new type system based on defined types
45 
46  // create all type symbols in a first step
47  for (const auto& cur : program.getTypes()) {
48  // support faulty codes with multiple definitions
49  if (env.isType(cur->getQualifiedName())) {
50  continue;
51  }
52 
53  // create type within type environment
54  if (auto* t = dynamic_cast<const AstSubsetType*>(cur)) {
55  env.createSubsetType(cur->getQualifiedName(), t->getTypeAttribute());
56  } else if (dynamic_cast<const AstUnionType*>(cur) != nullptr) {
57  // initialize the union
58  env.createUnionType(cur->getQualifiedName());
59  } else if (dynamic_cast<const AstRecordType*>(cur) != nullptr) {
60  // initialize the record
61  env.createRecordType(cur->getQualifiedName());
62  } else {
63  std::cout << "Unsupported type construct: " << typeid(cur).name() << "\n";
64  assert(false && "Unsupported Type Construct!");
65  }
66  }
67 
68  // link symbols in a second step
69  for (const auto& cur : program.getTypes()) {
70  Type* type = env.getModifiableType(cur->getQualifiedName());
71  assert(type && "It should be there!");
72 
73  if (dynamic_cast<const AstSubsetType*>(cur) != nullptr) {
74  // nothing to do here
75  } else if (auto* t = dynamic_cast<const AstUnionType*>(cur)) {
76  // get type as union type
77  auto* ut = dynamic_cast<UnionType*>(type);
78  if (ut == nullptr) {
79  continue; // support faulty input
80  }
81 
82  // add element types
83  for (const auto& cur : t->getTypes()) {
84  if (env.isType(cur)) {
85  ut->add(env.getType(cur));
86  }
87  }
88  } else if (auto* t = dynamic_cast<const AstRecordType*>(cur)) {
89  // get type as record type
90  auto* rt = dynamic_cast<RecordType*>(type);
91  if (rt == nullptr) {
92  continue; // support faulty input
93  }
94 
95  // add fields
96  for (const auto& f : t->getFields()) {
97  if (env.isType(f.type)) {
98  rt->add(f.name, env.getType(f.type));
99  }
100  }
101  } else {
102  std::cout << "Unsupported type construct: " << typeid(cur).name() << "\n";
103  assert(false && "Unsupported Type Construct!");
104  }
105  }
106 
107  // Assign types to unions.
108  Graph<AstQualifiedName> typeDependencyGraph;
109  for (const auto& cur : program.getTypes()) {
110  if (auto type = dynamic_cast<const AstSubsetType*>(cur)) {
111  switch (type->getTypeAttribute()) {
112  case TypeAttribute::Signed:
113  typeDependencyGraph.insert(type->getQualifiedName(), "number");
114  break;
115  case TypeAttribute::Unsigned:
116  typeDependencyGraph.insert(type->getQualifiedName(), "unsigned");
117  break;
118  case TypeAttribute::Float:
119  typeDependencyGraph.insert(type->getQualifiedName(), "float");
120  break;
121  case TypeAttribute::Symbol:
122  typeDependencyGraph.insert(type->getQualifiedName(), "symbol");
123  break;
124  case TypeAttribute::Record:
125  assert(false && "invalid type");
126  }
127  } else if (dynamic_cast<const AstRecordType*>(cur) != nullptr) {
128  // do nothing
129  } else if (auto type = dynamic_cast<const AstUnionType*>(cur)) {
130  for (const auto& subtype : type->getTypes()) {
131  typeDependencyGraph.insert(type->getQualifiedName(), subtype);
132  }
133  } else {
134  std::cout << "Unsupported type construct: " << typeid(cur).name() << "\n";
135  assert(false && "Unsupported Type Construct!");
136  }
137  }
138  for (const auto& cur : program.getTypes()) {
139  if (auto unionType = dynamic_cast<const AstUnionType*>(cur)) {
140  AstQualifiedName unionName = unionType->getQualifiedName();
141 
142  auto itereratorToUnion = unionTypes.find(unionName);
143 
144  // Initialize with the empty set
145  if (itereratorToUnion == unionTypes.end()) {
146  itereratorToUnion = unionTypes.insert({unionName, {}}).first;
147  }
148 
149  auto& associatedTypes = itereratorToUnion->second;
150 
151  // Insert any reachable predefined type
152  if (typeDependencyGraph.reaches(unionName, "number")) {
153  associatedTypes.insert(TypeAttribute::Signed);
154  }
155 
156  if (typeDependencyGraph.reaches(unionName, "symbol")) {
157  associatedTypes.insert(TypeAttribute::Symbol);
158  }
159 
160  if (typeDependencyGraph.reaches(unionName, "unsigned")) {
161  associatedTypes.insert(TypeAttribute::Unsigned);
162  }
163 
164  if (typeDependencyGraph.reaches(unionName, "float")) {
165  associatedTypes.insert(TypeAttribute::Float);
166  }
167  }
168  }
169 }
