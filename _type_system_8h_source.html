173 /**
174  * A collection to represent sets of types. In addition to ordinary set capabilities
175  * it may also represent the set of all types -- without being capable of iterating over those.
176  *
177  * It is the basic entity to conduct sub- and super-type computations.
178  */
179 struct TypeSet {
180 public:
181  using const_iterator = IterDerefWrapper<typename std::set<const Type*>::const_iterator>;
182 
183  // -- constructors, destructors and assignment operations --
184 
185  TypeSet(bool all = false) : all(all) {}
186 
187  TypeSet(const TypeSet& other) = default;
188 
189  TypeSet(TypeSet&& other) noexcept : all(other.all), types() {
190  types.swap(other.types);
191  }
192 
193  template <typename... Types>
194  explicit TypeSet(const Types&... types) : all(false) {
195  for (const Type* cur : toVector<const Type*>(&types...)) {
196  this->types.insert(cur);
197  }
198  }
199 
200  TypeSet& operator=(const TypeSet& other) = default;
201 
202  /** A factory function for the all-types set */
203  static TypeSet getAllTypes() {
204  return TypeSet(true);
205  }
206 
207  /** Emptiness check */
208  bool empty() const {
209  return !all && types.empty();
210  }
211 
212  /** Universality check */
213  bool isAll() const {
214  return all;
215  }
216 
217  /** Determines the size of this set unless it is the universal set */
218  std::size_t size() const {
219  assert(!all && "Unable to give size of universe.");
220  return types.size();
221  }
222 
223  /** Determines whether a given type is included or not */
224  bool contains(const Type& type) const {
225  return all || types.find(&type) != types.end();
226  }
227 
228  /** Adds the given type to this set */
229  void insert(const Type& type) {
230  if (!all) {
231  types.insert(&type);
232  }
233  }
234 
235  /** Calculate intersection of two TypeSet */
236  static TypeSet intersection(const TypeSet& left, const TypeSet& right) {
237  TypeSet result;
238 
239  if (left.isAll()) {
240  return right;
241  } else if (right.isAll()) {
242  return left;
243  }
244 
245  for (const auto& element : left) {
246  if (right.contains(element)) {
247  result.insert(element);
248  }
249  }
250 
251  return result;
252  }
253 
254  /** Inserts all the types of the given set into this set */
255  void insert(const TypeSet& set) {
256  if (all) {
257  return;
258  }
259 
260  // if the other set is universal => make this one universal
261  if (set.isAll()) {
262  all = true;
263  types.clear();
264  return;
265  }
266 
267  // add types one by one
268  for (const auto& t : set) {
269  insert(t);
270  }
271  }
272 
273  /** Allows to iterate over the types contained in this set (only if not universal) */
274  const_iterator begin() const {
275  assert(!all && "Unable to enumerate universe.");
276  return derefIter(types.begin());
277  }
278 
279  /** Allows to iterate over the types contained in this set (only if not universal) */
280  const_iterator end() const {
281  assert(!all && "Unable to enumerate universe.");
282  return derefIter(types.end());
283  }
284 
285  /** Determines whether this set is a subset of the given set */
286  bool isSubsetOf(const TypeSet& b) const {
287  if (all) {
288  return b.isAll();
289  }
290  return all_of(*this, [&](const Type& cur) { return b.contains(cur); });
291  }
292 
293  /** Determines equality between type sets */
294  bool operator==(const TypeSet& other) const {
295  return all == other.all && types == other.types;
296  }
297 
298  /** Determines inequality between type sets */
299  bool operator!=(const TypeSet& other) const {
300  return !(*this == other);
301  }
302 
303  /** Adds print support for type sets */
304  void print(std::ostream& out) const {
305  if (all) {
306  out << "{ - all types - }";
307  } else {
308  out << "{"
309  << join(types, ",", [](std::ostream& out, const Type* type) { out << type->getName(); })
310  << "}";
311  }
312  }
313 
314  friend std::ostream& operator<<(std::ostream& out, const TypeSet& set) {
315  set.print(out);
316  return out;
317  }
318 
319 private:
320  /** True if it is the all-types set, false otherwise */
321  bool all;
322 
323  /** The enumeration of types in case it is not the all-types set */
324  std::set<const Type*, deref_less<Type>> types;
325 };

460 /**
461  * Determine if a type analysis' result is equivalent to the given TypeAttribute.
462  */
463 template <typename T> // T = Type or T = Typeset
464 bool eqTypeTypeAttribute(const TypeAttribute ramType, const T& type) {
465  switch (ramType) {
466  case TypeAttribute::Signed:
467  return isNumberType(type);
468  case TypeAttribute::Unsigned:
469  return isUnsignedType(type);
470  case TypeAttribute::Float:
471  return isFloatType(type);
472  case TypeAttribute::Symbol:
473  return isSymbolType(type);
474  case TypeAttribute::Record:
475  return isRecordType(type);
476  }
477  assert(false && "unhandled `TypeAttribute`");
478  exit(EXIT_FAILURE);
479 }

481 /**
482  * Convert a type analysis' type/set of type to the the TypeAttribute
483  */
484 template <typename T> // T = Type or T = Typeset
485 TypeAttribute getTypeAttribute(const T& type) {
486  TypeAttribute primitiveType;
487  if (isNumberType(type)) {
488  primitiveType = TypeAttribute::Signed;
489  } else if (isUnsignedType(type)) {
490  primitiveType = TypeAttribute::Unsigned;
491  } else if (isFloatType(type)) {
492  primitiveType = TypeAttribute::Float;
493  } else if (isRecordType(type)) {
494  primitiveType = TypeAttribute::Record;
495  } else if (isSymbolType(type)) {
496  primitiveType = TypeAttribute::Symbol;
497  } else {
498  std::cerr << "Unknown type class" << std::endl;
499  std::exit(EXIT_FAILURE);
500  }
501  return primitiveType;
502 }
