84  std::map<int, const AstExecutionOrder*> result;

87  }
88  return result;
89  }
90 
91  AstExecutionPlan* clone() const override {
92  auto res = new AstExecutionPlan();
93  res->setSrcLoc(getSrcLoc());
94  for (auto& plan : plans) {
95  res->setOrderFor(plan.first, std::unique_ptr<AstExecutionOrder>(plan.second->clone()));
96  }
97  return res;
98  }
99 
100  void apply(const AstNodeMapper& map) override {
101  for (auto& plan : plans) {
102  plan.second = map(std::move(plan.second));
103  }
104  }
105 
106  std::vector<const AstNode*> getChildNodes() const override {
107  std::vector<const AstNode*> childNodes;
108  for (auto& plan : plans) {
109  childNodes.push_back(plan.second.get());
110  }
111  return childNodes;
112  }
113 
114 protected:
115  void print(std::ostream& out) const override {
116  if (!plans.empty()) {
117  out << " .plan ";
118  out << join(plans, ", ",
119  [](std::ostream& os, const auto& arg) { os << arg.first << ":" << *arg.second; });
120  }
121  }
122 
123  bool equal(const AstNode& node) const override {
124  const auto& other = static_cast<const AstExecutionPlan&>(node);
125  return equal_targets(plans, other.plans);
126  }
127 
128 private:
129  /** mapping versions of clauses to execution plans */
130  std::map<int, std::unique_ptr<AstExecutionOrder>> plans;

133 /**
134  * Intermediate representation of a datalog clause.
135  *
136  * A clause can either be:
137  * - a fact - a clause with no body (e.g., X(a,b))
138  * - a rule - a clause with a head and a body (e.g., Y(a,b) -: X(a,b))
139  *
140  * TODO (azreika): make clause abstract and split into two subclasses: Rule and Fact
141  */
142 class AstClause : public AstNode {
143 public:
144  AstClause() = default;
145  AstClause(std::unique_ptr<AstAtom> head, std::vector<std::unique_ptr<AstLiteral>> bodyLiterals,
146  std::unique_ptr<AstExecutionPlan> plan)
147  : AstNode(), head(std::move(head)), bodyLiterals(std::move(bodyLiterals)),
148  plan(std::move(plan)){};
149 
150  /** Add a Literal to the body of the clause */
151  void addToBody(std::unique_ptr<AstLiteral> literal) {
152  bodyLiterals.push_back(std::move(literal));
153  }
154 
155  /** Set the head of clause to @p h */
156  void setHead(std::unique_ptr<AstAtom> h) {
157  head = std::move(h);
158  }
159 
160  /** Set the bodyLiterals of clause to @p body */
161  void setBodyLiterals(std::vector<std::unique_ptr<AstLiteral>> body) {
162  bodyLiterals = std::move(body);
163  }
164 
165  /** Return the atom that represents the head of the clause */
166  AstAtom* getHead() const {
167  return head.get();
168  }
169 
170  /** Obtains a copy of the internally maintained body literals */
171  std::vector<AstLiteral*> getBodyLiterals() const {
172  return toPtrVector(bodyLiterals);
173  }
174 
175  /** Obtains the execution plan associated to this clause or null if there is none */
176  const AstExecutionPlan* getExecutionPlan() const {
177  return plan.get();
178  }
179 
180  /** Updates the execution plan associated to this clause */
181  void setExecutionPlan(std::unique_ptr<AstExecutionPlan> plan) {
182  this->plan = std::move(plan);
183  }
184 
185  /** Resets the execution plan */
186  void clearExecutionPlan() {
187  plan = nullptr;
188  }
189 
190  AstClause* clone() const override {
191  auto res = new AstClause();
192  res->setSrcLoc(getSrcLoc());
193  if (getExecutionPlan() != nullptr) {
194  res->setExecutionPlan(std::unique_ptr<AstExecutionPlan>(plan->clone()));
195  }
196  res->head = souffle::clone(head);
197  for (const auto& lit : bodyLiterals) {
198  res->bodyLiterals.emplace_back(lit->clone());
199  }
200  return res;
201  }
202 
203  void apply(const AstNodeMapper& map) override {
204  head = map(std::move(head));
205  for (auto& lit : bodyLiterals) {
206  lit = map(std::move(lit));
207  }
208  }
209 
210  std::vector<const AstNode*> getChildNodes() const override {
211  std::vector<const AstNode*> res = {head.get()};
212  for (auto& cur : bodyLiterals) {
213  res.push_back(cur.get());
214  }
215  return res;
216  }
217 
218 protected:
219  void print(std::ostream& os) const override {
220  if (head != nullptr) {
221  os << *head;
222  }
223  if (!bodyLiterals.empty()) {
224  os << " :- \n ";
225  os << join(getBodyLiterals(), ",\n ", print_deref<AstLiteral*>());
226  }
227  os << ".";
228  if (plan != nullptr) {
229  os << *plan;
230  }
231  }
232 
233  bool equal(const AstNode& node) const override {
234  const auto& other = static_cast<const AstClause&>(node);
235  return equal_ptr(head, other.head) && equal_targets(bodyLiterals, other.bodyLiterals) &&
236  equal_ptr(plan, other.plan);
237  }
238 
239  /** head of the clause */
240  std::unique_ptr<AstAtom> head;
241 
242  /** body literals of clause */
243  std::vector<std::unique_ptr<AstLiteral>> bodyLiterals;
244 
245  /** user defined execution plan (if not defined, plan is null) */
246  std::unique_ptr<AstExecutionPlan> plan;
247 };
