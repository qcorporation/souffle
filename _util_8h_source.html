113 /**
114  * Converts a string to a RamSigned
115  *
116  * This procedure has similar behaviour to std::stoi/stoll.
117  *
118  * The procedure accepts prefixes 0b (if base = 2) and 0x (if base = 16)
119  * If base = 0, the procedure will try to infer the base from the prefix, if present.
120  */
121 inline RamSigned RamSignedFromString(
122  const std::string& str, std::size_t* position = nullptr, const int base = 10) {
123  RamSigned val;
124 
125  if (base == 0) {
126  if (isPrefix("-0b", str) || isPrefix("0b", str)) {
127  return RamSignedFromString(str, position, 2);
128  } else if (isPrefix("-0x", str) || isPrefix("0x", str)) {
129  return RamSignedFromString(str, position, 16);
130  } else {
131  return RamSignedFromString(str, position);
132  }
133  }
134  std::string binaryNumber;
135  bool parsingBinary = base == 2;
136 
137  // stoi/stoll can't handle base 2 prefix by default.
138  if (parsingBinary) {
139  if (isPrefix("-0b", str)) {
140  binaryNumber = "-" + str.substr(3);
141  } else if (isPrefix("0b", str)) {
142  binaryNumber = str.substr(2);
143  }
144  }
145  const std::string& tmp = parsingBinary ? binaryNumber : str;
146 
147 #if RAM_DOMAIN_SIZE == 64
148  val = std::stoll(tmp, position, base);
149 #else
150  val = std::stoi(tmp, position, base);
151 #endif
152 
153  if (parsingBinary && position != nullptr) {
154  *position += 2;
155  }
156 
157  return val;
158 }

172 /**
173  * Converts a string to a RamUnsigned
174  *
175  * This procedure has similar behaviour to std::stoul/stoull.
176  *
177  * The procedure accepts prefixes 0b (if base = 2) and 0x (if base = 16)
178  * If base = 0, the procedure will try to infer the base from the prefix, if present.
179  */
180 inline RamUnsigned RamUnsignedFromString(
181  const std::string& str, std::size_t* position = nullptr, const int base = 10) {
182  // Be default C++ (stoul) allows unsigned numbers starting with "-".
183  if (isPrefix("-", str)) {
184  throw std::invalid_argument("Unsigned number can't start with minus.");
185  }
186 
187  if (base == 0) {
188  if (isPrefix("0b", str)) {
189  return RamUnsignedFromString(str, position, 2);
190  } else if (isPrefix("0x", str)) {
191  return RamUnsignedFromString(str, position, 16);
192  } else {
193  return RamUnsignedFromString(str, position);
194  }
195  }
196 
197  // stoul/stoull can't handle binary prefix by default.
198  std::string binaryNumber;
199  bool parsingBinary = false;
200  if (base == 2 && isPrefix("0b", str)) {
201  binaryNumber = str.substr(2);
202  parsingBinary = true;
203  }
204  const std::string& tmp = parsingBinary ? binaryNumber : str;
205 
206  RamUnsigned val;
207 #if RAM_DOMAIN_SIZE == 64
208  val = std::stoull(tmp, position, base);
209 #else
210  val = std::stoul(tmp, position, base);
211 #endif
212 
213  if (parsingBinary && position != nullptr) {
214  *position += 2;
215  }
216 
217  // check if it's safe to cast (stoul returns unsigned long)
218  if (val > std::numeric_limits<RamUnsigned>::max()) {
219  throw std::invalid_argument("Unsigned number of of bounds");
220  }
221 
222  return static_cast<RamUnsigned>(val);
223 }

375 /**
376  * A utility class enabling representation of ranges by pairing
377  * two iterator instances marking lower and upper boundaries.
378  */
379 template <typename Iter>
380 struct range {
381  // the lower and upper boundary
382  Iter a, b;
383 
384  // a constructor accepting a lower and upper boundary
385  range(Iter a, Iter b) : a(std::move(a)), b(std::move(b)) {}
386 
387  // default copy / move and assignment support
388  range(const range&) = default;
389  range(range&&) = default;
390  range& operator=(const range&) = default;
391 
392  // get the lower boundary (for for-all loop)
393  Iter& begin() {
394  return a;
395  }
396  const Iter& begin() const {
397  return a;
398  }
399 
400  // get the upper boundary (for for-all loop)
401  Iter& end() {
402  return b;
403  }
404  const Iter& end() const {
405  return b;
406  }
407 
408  // emptiness check
409  bool empty() const {
410  return a == b;
411  }
412 
413  // splits up this range into the given number of partitions
414  std::vector<range> partition(int np = 100) {
415  // obtain the size
416  int n = 0;
417  for (auto i = a; i != b; ++i) {
418  n++;
419  }
420 
421  // split it up
422  auto s = n / np;
423  auto r = n % np;
424  std::vector<range> res;
425  res.reserve(np);
426  auto cur = a;
427  auto last = cur;
428  int i = 0;
429  int p = 0;
430  while (cur != b) {
431  ++cur;
432  i++;
433  if (i >= (s + (p < r ? 1 : 0))) {
434  res.push_back({last, cur});
435  last = cur;
436  p++;
437  i = 0;
438  }
439  }
440  if (cur != last) {
441  res.push_back({last, cur});
442  }
443  return res;
444  }
445 };

600  * A auxiliary class to be returned by the join function aggregating the information
601  * required to print a list of elements as well as the implementation of the printing
602  * itself.
603  */
604 template <typename Iter, typename Printer>
605 class joined_sequence {
606  /** The begin of the range to be printed */
607  Iter begin;
608 
609  /** The end of the range to be printed */
610  Iter end;
611 
612  /** The seperator to be utilized between elements */
613  std::string sep;
614 
615  /** A functor printing an element */
616  Printer p;
617 
618 public:
619  /** A constructor setting up all fields of this class */
620  joined_sequence(const Iter& a, const Iter& b, std::string sep, Printer p)
621  : begin(a), end(b), sep(std::move(sep)), p(std::move(p)) {}
622 
623  /** The actual print method */
624  friend std::ostream& operator<<(std::ostream& out, const joined_sequence& s) {
625  auto cur = s.begin;
626  if (cur == s.end) {
627  return out;
628  }
629 
630  s.p(out, *cur);
631  ++cur;
632  for (; cur != s.end; ++cur) {
633  out << s.sep;
634  s.p(out, *cur);
635  }
636  return out;
637  }

1394 /**
1395  * An Least-Recently-Used cache for arbitrary element types. Elements can be signaled
1396  * to be accessed and iterated through in their LRU order.
1397  */
1398 template <typename T, unsigned size = 1>
1399 class LRUCache {
1400  // the list of pointers maintained
1401  std::array<T, size> entries;
1402 
1403  // pointer to predecessor / successor in the entries list
1404  std::array<std::size_t, size> priv; // < predecessor of element i
1405  std::array<std::size_t, size> post; // < successor of element i
1406 
1407  std::size_t first{0}; // < index of the first element
1408  std::size_t last{size - 1}; // < index of the last element
1409 
1410 public:
1411  // creates a new, empty cache
1412  LRUCache(const T& val = T()) {
1413  for (unsigned i = 0; i < size; i++) {
1414  entries[i] = val;
1415  priv[i] = i - 1;
1416  post[i] = i + 1;
1417  }
1418  priv[first] = last;
1419  post[last] = first;
1420  }
1421 
1422  // clears the content of this cache
1423  void clear(const T& val = T()) {
1424  for (auto& cur : entries) {
1425  cur = val;
1426  }
1427  }
1428 
1429  // registers an access to the given element
1430  void access(const T& val) {
1431  // test whether it is contained
1432  for (std::size_t i = 0; i < size; i++) {
1433  if (entries[i] != val) {
1434  continue;
1435  }
1436 
1437  // -- move this one to the front --
1438 
1439  // if it is the first, nothing to handle
1440  if (i == first) {
1441  return;
1442  }
1443 
1444  // if this is the last, just first and last need to change
1445  if (i == last) {
1446  auto tmp = last;
1447  last = priv[last];
1448  first = tmp;
1449  return;
1450  }
1451 
1452  // otherwise we need to update the linked list
1453 
1454  // remove from current position
1455  post[priv[i]] = post[i];
1456  priv[post[i]] = priv[i];
1457 
1458  // insert in first position
1459  post[i] = first;
1460  priv[i] = last;
1461  priv[first] = i;
1462  post[last] = i;
1463 
1464  // update first pointer
1465  first = i;
1466  return;
1467  }
1468  // not present => drop last, make it first
1469  entries[last] = val;
1470  auto tmp = last;
1471  last = priv[last];
1472  first = tmp;
1473  }
1474 
1475  /**
1476  * Iterates over the elements within this cache in LRU order.
1477  * The operator is applied on each element. If the operation
1478  * returns false, iteration is continued. If the operator return
