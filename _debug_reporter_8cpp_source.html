31 bool DebugReporter::transform(AstTranslationUnit& translationUnit) {
32  translationUnit.getDebugReport().startSection();
33  std::stringstream datalogSpecOriginal;
34  datalogSpecOriginal << *translationUnit.getProgram();
35  auto start = std::chrono::high_resolution_clock::now();
36  bool changed = applySubtransformer(translationUnit, wrappedTransformer.get());
37  auto end = std::chrono::high_resolution_clock::now();
38  std::string runtimeStr = "(" + std::to_string(std::chrono::duration<double>(end - start).count()) + "s)";
39  if (changed) {
40  generateDebugReport(translationUnit, datalogSpecOriginal.str());
41  translationUnit.getDebugReport().endSection(
42  wrappedTransformer->getName(), wrappedTransformer->getName() + " " + runtimeStr);
43  } else {
44  translationUnit.getDebugReport().endSection(wrappedTransformer->getName(),
45  wrappedTransformer->getName() + " " + runtimeStr + " (unchanged)");
46  }
47  return changed;
48 }

50 DebugReportSection formatCodeSection(const std::string& id, const std::string& title, std::string code) {
51  std::stringstream codeHTML;
