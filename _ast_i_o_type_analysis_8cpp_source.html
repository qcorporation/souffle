25 void IOType::run(const AstTranslationUnit& translationUnit) {
26  const AstProgram& program = *translationUnit.getProgram();
27  visitDepthFirst(program, [&](const AstIO& io) {
28  auto* relation = getRelation(program, io.getQualifiedName());
29  if (relation == nullptr) {
30  return;
31  }
32  switch (io.getType()) {
33  case AstIO::InputIO:
34  inputRelations.insert(relation);
35  break;
36  case AstIO::OutputIO:
37  outputRelations.insert(relation);
38  break;
39  case AstIO::PrintsizeIO:
40  printSizeRelations.insert(relation);
41  outputRelations.insert(relation);
42  break;
43  default:
44  assert("Unrecognized I/O operation");
45  }
46  });
47 }
