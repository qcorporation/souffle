99 // Compile time sanity checks
100 static_assert(std::is_integral<RamSigned>::value && std::is_signed<RamSigned>::value,
101  "RamSigned must be represented by a signed type.");
102 static_assert(std::is_integral<RamUnsigned>::value && !std::is_signed<RamUnsigned>::value,
103  "RamUnsigned must be represented by an unsigned type.");
104 static_assert(std::is_floating_point<RamFloat>::value && sizeof(RamFloat) * 8 == RAM_DOMAIN_SIZE,
105  "RamFloat must be represented by a floating point and have the same size as other types.");

107 template <typename T>
108 constexpr bool isRamType = (std::is_same<T, RamDomain>::value || std::is_same<T, RamSigned>::value ||
109  std::is_same<T, RamUnsigned>::value || std::is_same<T, RamFloat>::value);

116 /** Cast a type by reinterpreting its bits. Domain is restricted to Ram Types only.
117  * Template takes two types (second type is never necessary because it can be deduced from the argument)
118  * The following always holds
119  * For type T and a : T
120  * ramBitCast<T>(ramBitCast<RamDomain>(a)) == a
121  **/
122 template <typename To = RamDomain, typename From>
123 To ramBitCast(From source) {
124  static_assert(isRamType<From> && isRamType<To>, "Bit casting should only be used on Ram Types.");
125  static_assert(sizeof(To) == sizeof(From), "Can't bit cast types with different size.");
126  To destination;
127  memcpy(&destination, &source, sizeof(destination));
128  return destination;
129 }
