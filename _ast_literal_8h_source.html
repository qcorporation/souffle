43 /**
44  * Subclass of Literal that represents the use of a relation
45  * either in the head or in the body of a Clause, e.g., parent(x,y).
46  * The arguments of the atom can be variables or constants.
47  */
48 class AstAtom : public AstLiteral {
49 public:
50  AstAtom(AstQualifiedName name = AstQualifiedName()) : name(std::move(name)) {}
51 
52  AstAtom(AstQualifiedName name, std::vector<std::unique_ptr<AstArgument>> args, SrcLocation srcLoc)
53  : name(std::move(name)), arguments(std::move(args)) {
54  setSrcLoc(srcLoc);
55  }
56 
57  /** get qualified name */
58  const AstQualifiedName& getQualifiedName() const {
59  return name;
60  }
61 
62  /** get arity of the atom */
63  size_t getArity() const {
64  return arguments.size();
65  }
66 
67  /** set qualified name */
68  void setQualifiedName(const AstQualifiedName& n) {
69  name = n;
70  }
71 
72  /** add argument to the atom */
73  void addArgument(std::unique_ptr<AstArgument> arg) {
74  arguments.push_back(std::move(arg));
75  }
76 
77  /** get arguments */
78  std::vector<AstArgument*> getArguments() const {
79  return toPtrVector(arguments);
80  }
81 
82  AstAtom* clone() const override {
83  auto res = new AstAtom(name);
84  res->setSrcLoc(getSrcLoc());
85  for (const auto& cur : arguments) {
86  res->arguments.emplace_back(cur->clone());
87  }
88  return res;
89  }
90 
91  void apply(const AstNodeMapper& map) override {
92  for (auto& arg : arguments) {
93  arg = map(std::move(arg));
94  }
95  }
96 
97  std::vector<const AstNode*> getChildNodes() const override {
98  std::vector<const AstNode*> res;
99  for (auto& cur : arguments) {
100  res.push_back(cur.get());
101  }
102  return res;
103  }
104 
105 protected:
106  void print(std::ostream& os) const override {
107  os << getQualifiedName() << "(";
108  os << join(arguments, ",", print_deref<std::unique_ptr<AstArgument>>());
109  os << ")";
110  }
111 
112  bool equal(const AstNode& node) const override {
113  const auto& other = static_cast<const AstAtom&>(node);
114  return name == other.name && equal_targets(arguments, other.arguments);
115  }
116 
117  /** name */
118  AstQualifiedName name;
119 
120  /** arguments */
121  std::vector<std::unique_ptr<AstArgument>> arguments;
122 };

251  }
252

255  operation = op;
256  }
257 
258  AstBinaryConstraint* clone() const override {
259  auto* res = new AstBinaryConstraint(operation, std::unique_ptr<AstArgument>(lhs->clone()),
260  std::unique_ptr<AstArgument>(rhs->clone()));
261  res->setSrcLoc(getSrcLoc());
262  return res;
263  }
264 
265  void apply(const AstNodeMapper& map) override {
266  lhs = map(std::move(lhs));
267  rhs = map(std::move(rhs));
268  }
269 
270  std::vector<const AstNode*> getChildNodes() const override {
271  return {lhs.get(), rhs.get()};
272  }
273 
274 protected:
275  void print(std::ostream& os) const override {
276  os << *lhs;
277  os << " " << toBinaryConstraintSymbol(operation) << " ";
278  os << *rhs;
279  }
280 
281  bool equal(const AstNode& node) const override {
282  assert(nullptr != dynamic_cast<const AstBinaryConstraint*>(&node));
283  const auto& other = static_cast<const AstBinaryConstraint&>(node);
284  return operation == other.operation && equal_ptr(lhs, other.lhs) && equal_ptr(rhs, other.rhs);
285  }
286

292

294  std::unique_ptr<AstArgument> rhs;
295 };
