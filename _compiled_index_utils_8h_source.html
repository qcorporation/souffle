74 // -------- generic tuple comparator ----------
75 
76 template <unsigned... Columns>
77 struct comparator;
78 
79 template <unsigned First, unsigned... Rest>
80 struct comparator<First, Rest...> {
81  template <typename T>
82  int operator()(const T& a, const T& b) const {
83  return (a[First] < b[First]) ? -1 : ((a[First] > b[First]) ? 1 : comparator<Rest...>()(a, b));
84  }
85  template <typename T>
86  bool less(const T& a, const T& b) const {
87  return a[First] < b[First] || (a[First] == b[First] && comparator<Rest...>().less(a, b));
88  }
89  template <typename T>
90  bool equal(const T& a, const T& b) const {
91  return a[First] == b[First] && comparator<Rest...>().equal(a, b);
92  }
93 };
94
